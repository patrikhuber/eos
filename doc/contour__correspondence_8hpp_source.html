159  static ContourLandmarks load(std::string filename)
160  {
161  // parse() as well as extracting the data can throw std::runtime error or toml::exception,
162  // so ideally you'd want to call this c'tor within a try-catch.
163  const auto data = toml::parse(filename);
164 
165  // Todo: Handle the case when '[contour_landmarks]' is empty - now it just throws. Use
166  // std::unordered_map::count.
167  const auto contour_table = toml::get<toml::Table>(data.at("contour_landmarks"));
168 
169  ContourLandmarks contour;
170  // There might be a vector<string> or a vector<int>, we need to check for that and convert to vector<string>.
171  // Todo: Again, check whether the key exists first.
172  // Read all the "right" contour landmarks:
173  const auto right_contour = toml::get<std::vector<toml::value>>(contour_table.at("right"));
174  for (const auto& landmark : right_contour)
175  {
176  std::string value;
177  switch (landmark.type())
178  {
179  case toml::value_t::Integer:
180  value = std::to_string(toml::get<int>(landmark));
181  break;
182  case toml::value_t::String:
183  value = toml::get<std::string>(landmark);
184  break;
185  default:
186  throw std::runtime_error("unexpected type : " + toml::stringize(landmark.type()));
187  }
188  contour.right_contour.push_back(value);
189  }
190  // Now the same for all the "left" contour landmarks:
191  const auto left_contour = toml::get<std::vector<toml::value>>(contour_table.at("left"));
192  for (const auto& landmark : left_contour)
193  {
194  std::string value;
195  switch (landmark.type())
196  {
197  case toml::value_t::Integer:
198  value = std::to_string(toml::get<int>(landmark));
199  break;
200  case toml::value_t::String:
201  value = toml::get<std::string>(landmark);
202  break;
203  default:
204  throw std::runtime_error("unexpected type : " + toml::stringize(landmark.type()));
205  }
206  contour.left_contour.push_back(value);
207  }

209  return contour;

273 inline std::pair<std::vector<std::string>, std::vector<int>>
274 select_contour(float yaw_angle, const ContourLandmarks& contour_landmarks, const ModelContour& model_contour, float frontal_range_threshold)
275 {
276  using std::begin;
277  using std::end;
278  std::vector<int> model_contour_indices;
279  std::vector<std::string> contour_landmark_identifiers;
280  if (yaw_angle >= -frontal_range_threshold) // positive yaw = subject looking to the left
281  {
282  // ==> we use the right cnt-lms
283  model_contour_indices.insert(end(model_contour_indices), begin(model_contour.right_contour),
284  end(model_contour.right_contour));
285  contour_landmark_identifiers.insert(end(contour_landmark_identifiers),
286  begin(contour_landmarks.right_contour),
287  end(contour_landmarks.right_contour));
288  }
289  if (yaw_angle <= frontal_range_threshold)
290  {
291  // ==> we use the left cnt-lms
292  model_contour_indices.insert(end(model_contour_indices), begin(model_contour.left_contour),
293  end(model_contour.left_contour));
294  contour_landmark_identifiers.insert(end(contour_landmark_identifiers),
295  begin(contour_landmarks.left_contour),
296  end(contour_landmarks.left_contour));
297  }
298  // Note there's an overlap between the angles - if a subject is between +- 7.5°, both contours get added.
299  return std::make_pair(contour_landmark_identifiers, model_contour_indices);
300 };

322 inline std::tuple<std::vector<Eigen::Vector2f>, std::vector<Eigen::Vector4f>, std::vector<int>>
323 get_nearest_contour_correspondences(const core::LandmarkCollection<Eigen::Vector2f>& landmarks,
324  const std::vector<std::string>& landmark_contour_identifiers,
325  const std::vector<int>& model_contour_indices, const core::Mesh& mesh,
326  const glm::mat4x4& view_model, const glm::mat4x4& ortho_projection,
327  const glm::vec4& viewport)
328 {
329  // These are the additional contour-correspondences we're going to find and then use!
330  std::vector<Eigen::Vector4f> model_points_cnt; // the points in the 3D shape model
331  std::vector<int> vertex_indices_cnt; // their vertex indices
332  std::vector<Eigen::Vector2f> image_points_cnt; // the corresponding 2D landmark points
333 
334  // For each 2D-CNT-LM, find the closest 3DMM-CNT-LM and add to correspondences:
335  // Note: If we were to do this for all 3DMM vertices, then ray-casting (i.e. glm::unproject) would be
336  // quicker to find the closest vertex)
337  for (const auto& ibug_idx : landmark_contour_identifiers)
338  {
339  // Check if the contour landmark is amongst the landmarks given to us (from detector or ground truth):
340  // (Note: Alternatively, we could filter landmarks beforehand and then just loop over landmarks =>
341  // means one less function param here. Separate filtering from actual algorithm.)
342  const auto result = std::find_if(begin(landmarks), end(landmarks), [&ibug_idx](auto&& e) {
343  return e.name == ibug_idx;
344  }); // => this can go outside the loop
345  if (result == std::end(landmarks))
346  {
347  continue; // This should be okay; So it's possible that the function will not return any
348  // correspondences.
349  }
350 
351  const auto screen_point_2d_contour_landmark = result->coordinates;
352 
353  std::vector<float> distances_2d;
354  for (auto model_contour_vertex_idx : model_contour_indices) // we could actually pre-project them,
355  // i.e. only project them once, not for
356  // each landmark newly...
357  {
358  const glm::vec3 vertex(mesh.vertices[model_contour_vertex_idx][0],
359  mesh.vertices[model_contour_vertex_idx][1],
360  mesh.vertices[model_contour_vertex_idx][2]);
361  const glm::vec3 proj = glm::project(vertex, view_model, ortho_projection, viewport);
362  const Eigen::Vector2f screen_point_model_contour(proj.x, proj.y);
363 
364  const double dist = (screen_point_model_contour - screen_point_2d_contour_landmark).norm();
365  distances_2d.emplace_back(dist);
366  }
367  const auto min_ele = std::min_element(begin(distances_2d), end(distances_2d));
368  // Todo: Cover the case when cnt_indices_to_use.size() is 0.
369  const auto min_ele_idx = std::distance(begin(distances_2d), min_ele);
370  const auto the_3dmm_vertex_id_that_is_closest = model_contour_indices[min_ele_idx];
371 
372  const Eigen::Vector4f vertex(mesh.vertices[the_3dmm_vertex_id_that_is_closest][0],
373  mesh.vertices[the_3dmm_vertex_id_that_is_closest][1],
374  mesh.vertices[the_3dmm_vertex_id_that_is_closest][2], 1.0f);
375  model_points_cnt.emplace_back(vertex);
376  vertex_indices_cnt.emplace_back(the_3dmm_vertex_id_that_is_closest);
377  image_points_cnt.emplace_back(screen_point_2d_contour_landmark);
378  }
379 
380  return std::make_tuple(image_points_cnt, model_points_cnt, vertex_indices_cnt);
381 };
