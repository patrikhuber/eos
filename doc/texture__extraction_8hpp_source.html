53 /* This function is copied from OpenCV, originally under BSD licence.
54  * imgwarp.cpp from OpenCV-3.2.0.
55  *
56  * Calculates coefficients of affine transformation
57  * which maps (xi,yi) to (ui,vi), (i=1,2,3):
58  *
59  * ui = c00*xi + c01*yi + c02
60  *
61  * vi = c10*xi + c11*yi + c12
62  *
63  * Coefficients are calculated by solving linear system:
64  * / x0 y0 1 0 0 0 \ /c00\ /u0\
65  * | x1 y1 1 0 0 0 | |c01| |u1|
66  * | x2 y2 1 0 0 0 | |c02| |u2|
67  * | 0 0 0 x0 y0 1 | |c10| |v0|
68  * | 0 0 0 x1 y1 1 | |c11| |v1|
69  * \ 0 0 0 x2 y2 1 / |c12| |v2|
70  *
71  * where:
72  * cij - matrix coefficients
73  */
74 // Note: The original functions used doubles.
75 inline Eigen::Matrix<float, 2, 3> get_affine_transform(const std::array<Eigen::Vector2f, 3>& src,
76  const std::array<Eigen::Vector2f, 3>& dst)
77 {
78  using Eigen::Matrix;
79  assert(src.size() == dst.size() && src.size() == 3);
80 
81  Matrix<float, 6, 6> A;
82  Matrix<float, 6, 1> b;
83 
84  for (int i = 0; i < 3; i++)
85  {
86  A.block<1, 2>(2 * i, 0) = src[i]; // the odd rows
87  A.block<1, 2>((2 * i) + 1, 3) = src[i]; // even rows
88  A(2 * i, 2) = 1.0f;
89  A((2 * i) + 1, 5) = 1.0f;
90  A.block<1, 3>(2 * i, 3).setZero();
91  A.block<1, 3>((2 * i) + 1, 0).setZero();
92  b.segment<2>(2 * i) = dst[i];
93  }
94 
95  Matrix<float, 6, 1> X = A.colPivHouseholderQr().solve(b);
96 
97  Matrix<float, 2, 3> transform_matrix;
98  transform_matrix.block<1, 3>(0, 0) = X.segment<3>(0);
99  transform_matrix.block<1, 3>(1, 0) = X.segment<3>(3);
100 
101  return transform_matrix;
102 };

184 inline core::Image4u
185 extract_texture(const core::Mesh& mesh, Eigen::Matrix<float, 3, 4> affine_camera_matrix,
186  const core::Image3u& image, const core::Image1d& depthbuffer, bool compute_view_angle = false,
187  TextureInterpolation mapping_type = TextureInterpolation::NearestNeighbour,
188  int isomap_resolution = 512)
189 {
190  assert(mesh.vertices.size() == mesh.texcoords.size());
191 
192  using Eigen::Vector2f;
193  using Eigen::Vector3f;
194  using Eigen::Vector4f;
195  using std::ceil;
196  using std::floor;
197  using std::max;
198  using std::min;
199  using std::pow;
200  using std::round;
201  using std::sqrt;
202 
203  Eigen::Matrix<float, 4, 4> affine_camera_matrix_with_z =
204  detail::calculate_affine_z_direction(affine_camera_matrix);
205 
206  // Todo: We should handle gray images, but output a 4-channel isomap nevertheless I think.
207  core::Image4u isomap(isomap_resolution, isomap_resolution); // We should initialise with zeros.
208  // Incidentially, the current Image4u c'tor
209  // does that.
210 
211  std::vector<std::future<void>> results;
212  for (const auto& triangle_indices : mesh.tvi)
213  {
214 
215  // Note: If there's a performance problem, there's no need to capture the whole mesh - we could
216  // capture only the three required vertices with their texcoords.
217  auto extract_triangle = [&mesh, &affine_camera_matrix_with_z, &triangle_indices, &depthbuffer,
218  &isomap, &mapping_type, &image, &compute_view_angle]() {
219 
220  // Find out if the current triangle is visible:
221  // We do a second rendering-pass here. We use the depth-buffer of the final image, and then, here,
222  // check if each pixel in a triangle is visible. If the whole triangle is visible, we use it to
223  // extract the texture.
224  // Possible improvement: - If only part of the triangle is visible, split it
225 
226  // This could be optimized in 2 ways though:
227  // - Use render(), or as in render(...), transfer the vertices once, not in a loop over all
228  // triangles (vertices are getting transformed multiple times)
229  // - We transform them later (below) a second time. Only do it once.
230 
231  const Vector4f v0_as_Vector4f(mesh.vertices[triangle_indices[0]][0],
232  mesh.vertices[triangle_indices[0]][1],
233  mesh.vertices[triangle_indices[0]][2], 1.0f);
234  const Vector4f v1_as_Vector4f(mesh.vertices[triangle_indices[1]][0],
235  mesh.vertices[triangle_indices[1]][1],
236  mesh.vertices[triangle_indices[1]][2], 1.0f);
237  const Vector4f v2_as_Vector4f(mesh.vertices[triangle_indices[2]][0],
238  mesh.vertices[triangle_indices[2]][1],
239  mesh.vertices[triangle_indices[2]][2], 1.0f);
240 
241  // Project the triangle vertices to screen coordinates, and use the depthbuffer to check whether
242  // the triangle is visible:
243  const Vector4f v0 = affine_camera_matrix_with_z * v0_as_Vector4f;
244  const Vector4f v1 = affine_camera_matrix_with_z * v1_as_Vector4f;
245  const Vector4f v2 = affine_camera_matrix_with_z * v2_as_Vector4f;
246 
247  if (!detail::is_triangle_visible(glm::tvec4<float>(v0[0], v0[1], v0[2], v0[3]),
248  glm::tvec4<float>(v1[0], v1[1], v1[2], v1[3]),
249  glm::tvec4<float>(v2[0], v2[1], v2[2], v2[3]), depthbuffer))
250  {
251  // continue;
252  return;
253  }
254 
255  float alpha_value;
256  if (compute_view_angle)
257  {
258  // Calculate how well visible the current triangle is:
259  // (in essence, the dot product of the viewing direction (0, 0, 1) and the face normal)
260  const Vector3f face_normal =
261  compute_face_normal(v0_as_Vector4f, v1_as_Vector4f, v2_as_Vector4f);
262  // Transform the normal to "screen" (kind of "eye") space using the upper 3x3 part of the
263  // affine camera matrix (=the translation can be ignored):
264  Vector3f face_normal_transformed =
265  affine_camera_matrix_with_z.block<3, 3>(0, 0) * face_normal;
266  face_normal_transformed.normalize(); // normalise to unit length
267  // Implementation notes regarding the affine camera matrix and the sign:
268  // If the matrix given were the model_view matrix, the sign would be correct.
269  // However, affine_camera_matrix includes glm::ortho, which includes a z-flip.
270  // So we need to flip one of the two signs.
271  // * viewing_direction(0.0f, 0.0f, 1.0f) is correct if affine_camera_matrix were only a model_view matrix
272  // * affine_camera_matrix includes glm::ortho, which flips z, so we flip the sign of viewing_direction.
273  // We don't need the dot product since viewing_direction.xy are 0 and .z is 1:
274  const float angle = -face_normal_transformed[2]; // flip sign, see above
275  assert(angle >= -1.f && angle <= 1.f);
276  // angle is [-1, 1].
277  // * +1 means 0° (same direction)
278  // * 0 means 90°
279  // * -1 means 180° (facing opposite directions)
280  // It's a linear relation, so +0.5 is 45° etc.
281  // An angle larger than 90° means the vertex won't be rendered anyway (because it's
282  // back-facing) so we encode 0° to 90°.
283  if (angle < 0.0f)
284  {
285  alpha_value = 0.0f;
286  } else
287  {
288  alpha_value = angle * 255.0f;
289  }
290  } else
291  {
292  // no visibility angle computation - if the triangle/pixel is visible, set the alpha chan to
293  // 255 (fully visible pixel).
294  alpha_value = 255.0f;
295  }
296 
297  // Todo: Documentation
298  std::array<Vector2f, 3> src_tri;
299  std::array<Vector2f, 3> dst_tri;
300 
301  Vector4f vec(mesh.vertices[triangle_indices[0]][0], mesh.vertices[triangle_indices[0]][1],
302  mesh.vertices[triangle_indices[0]][2], 1.0f);
303  Vector4f res = affine_camera_matrix_with_z * vec;
304  src_tri[0] = Vector2f(res[0], res[1]);
305 
306  vec = Vector4f(mesh.vertices[triangle_indices[1]][0], mesh.vertices[triangle_indices[1]][1],
307  mesh.vertices[triangle_indices[1]][2], 1.0f);
308  res = affine_camera_matrix_with_z * vec;
309  src_tri[1] = Vector2f(res[0], res[1]);
310 
311  vec = Vector4f(mesh.vertices[triangle_indices[2]][0], mesh.vertices[triangle_indices[2]][1],
312  mesh.vertices[triangle_indices[2]][2], 1.0f);
313  res = affine_camera_matrix_with_z * vec;
314  src_tri[2] = Vector2f(res[0], res[1]);
315 
316  dst_tri[0] = Vector2f((isomap.cols - 0.5) * mesh.texcoords[triangle_indices[0]][0],
317  (isomap.rows - 0.5) * mesh.texcoords[triangle_indices[0]][1]);
318  dst_tri[1] = Vector2f((isomap.cols - 0.5) * mesh.texcoords[triangle_indices[1]][0],
319  (isomap.rows - 0.5) * mesh.texcoords[triangle_indices[1]][1]);
320  dst_tri[2] = Vector2f((isomap.cols - 0.5) * mesh.texcoords[triangle_indices[2]][0],
321  (isomap.rows - 0.5) * mesh.texcoords[triangle_indices[2]][1]);
322 
323  // We now have the source triangles in the image and the source triangle in the isomap
324  // We use the inverse/ backward mapping approach, so we want to find the corresponding texel
325  // (texture-pixel) for each pixel in the isomap
326 
327  // Get the inverse Affine Transform from original image: from dst (pixel in isomap) to src (in
328  // image)
329  Eigen::Matrix<float, 2, 3> warp_mat_org_inv = get_affine_transform(dst_tri, src_tri);
330 
331  // We now loop over all pixels in the triangle and select, depending on the mapping type, the
332  // corresponding texel(s) in the source image
333  for (int x = min(dst_tri[0][0], min(dst_tri[1][0], dst_tri[2][0]));
334  x < max(dst_tri[0][0], max(dst_tri[1][0], dst_tri[2][0])); ++x)
335  {
336  for (int y = min(dst_tri[0][1], min(dst_tri[1][1], dst_tri[2][1]));
337  y < max(dst_tri[0][1], max(dst_tri[1][1], dst_tri[2][1])); ++y)
338  {
339  if (detail::is_point_in_triangle(Vector2f(x, y), dst_tri[0], dst_tri[1], dst_tri[2]))
340  {
341 
342  // As the coordinates of the transformed pixel in the image will most likely not lie
343  // on a texel, we have to choose how to calculate the pixel colors depending on the
344  // next texels
345 
346  // There are three different texture interpolation methods: area, bilinear and nearest
347  // neighbour
348 
349  // Area mapping: calculate mean color of texels in transformed pixel area
350  if (mapping_type == TextureInterpolation::Area)
351  {
352 
353  // calculate positions of 4 corners of pixel in image (src)
354  const Vector3f homogenous_dst_upper_left(x - 0.5f, y - 0.5f, 1.0f);
355  const Vector3f homogenous_dst_upper_right(x + 0.5f, y - 0.5f, 1.0f);
356  const Vector3f homogenous_dst_lower_left(x - 0.5f, y + 0.5f, 1.0f);
357  const Vector3f homogenous_dst_lower_right(x + 0.5f, y + 0.5f, 1.0f);
358 
359  const Vector2f src_texel_upper_left =
360  warp_mat_org_inv * homogenous_dst_upper_left;
361  const Vector2f src_texel_upper_right =
362  warp_mat_org_inv * homogenous_dst_upper_right;
363  const Vector2f src_texel_lower_left =
364  warp_mat_org_inv * homogenous_dst_lower_left;
365  const Vector2f src_texel_lower_right =
366  warp_mat_org_inv * homogenous_dst_lower_right;
367 
368  const float min_a = min(min(src_texel_upper_left[0], src_texel_upper_right[0]),
369  min(src_texel_lower_left[0], src_texel_lower_right[0]));
370  const float max_a = max(max(src_texel_upper_left[0], src_texel_upper_right[0]),
371  max(src_texel_lower_left[0], src_texel_lower_right[0]));
372  const float min_b = min(min(src_texel_upper_left[1], src_texel_upper_right[1]),
373  min(src_texel_lower_left[1], src_texel_lower_right[1]));
374  const float max_b = max(max(src_texel_upper_left[1], src_texel_upper_right[1]),
375  max(src_texel_lower_left[1], src_texel_lower_right[1]));
376 
377  Eigen::Vector3i color; // std::uint8_t actually.
378  int num_texels = 0;
379 
380  // loop over square in which quadrangle out of the four corners of pixel is
381  for (int a = ceil(min_a); a <= floor(max_a); ++a)
382  {
383  for (int b = ceil(min_b); b <= floor(max_b); ++b)
384  {
385  // check if texel is in quadrangle
386  if (detail::is_point_in_triangle(Vector2f(a, b), src_texel_upper_left,
387  src_texel_lower_left,
388  src_texel_upper_right) ||
389  detail::is_point_in_triangle(Vector2f(a, b), src_texel_lower_left,
390  src_texel_upper_right,
391  src_texel_lower_right))
392  {
393  if (a < image.cols && b < image.rows)
394  { // check if texel is in image
395  num_texels++;
396  color += Eigen::Vector3i(image(b, a)[0], image(b, a)[1],
397  image(b, a)[2]);
398  }
399  }
400  }
401  }
402  if (num_texels > 0)
403  color = color / num_texels;
404  else
405  { // if no corresponding texel found, nearest neighbour interpolation
406  // calculate corresponding position of dst_coord pixel center in image (src)
407  Vector3f homogenous_dst_coord(x, y, 1.0f);
408  Vector2f src_texel = warp_mat_org_inv * homogenous_dst_coord;
409 
410  if ((round(src_texel[1]) < image.rows) && round(src_texel[0]) < image.cols)
411  {
412  const int y = round(src_texel[1]);
413  const int x = round(src_texel[0]);
414  color = Eigen::Vector3i(image(y, x)[0], image(y, x)[1], image(y, x)[2]);
415  }
416  }
417  isomap(y, x) = {
418  static_cast<std::uint8_t>(color[0]), static_cast<std::uint8_t>(color[1]),
419  static_cast<std::uint8_t>(color[2]), static_cast<std::uint8_t>(alpha_value)};
420  }
421  // Bilinear mapping: calculate pixel color depending on the four neighbouring texels
422  else if (mapping_type == TextureInterpolation::Bilinear)
423  {
424 
425  // calculate corresponding position of dst_coord pixel center in image (src)
426  const Vector3f homogenous_dst_coord(x, y, 1.0f);
427  const Vector2f src_texel = warp_mat_org_inv * homogenous_dst_coord;
428 
429  // calculate euclidean distances to next 4 texels
430  float distance_upper_left = sqrt(pow(src_texel[0] - floor(src_texel[0]), 2) +
431  pow(src_texel[1] - floor(src_texel[1]), 2));
432  float distance_upper_right = sqrt(pow(src_texel[0] - floor(src_texel[0]), 2) +
433  pow(src_texel[1] - ceil(src_texel[1]), 2));
434  float distance_lower_left = sqrt(pow(src_texel[0] - ceil(src_texel[0]), 2) +
435  pow(src_texel[1] - floor(src_texel[1]), 2));
436  float distance_lower_right = sqrt(pow(src_texel[0] - ceil(src_texel[0]), 2) +
437  pow(src_texel[1] - ceil(src_texel[1]), 2));
438 
439  // normalise distances that the sum of all distances is 1
440  const float sum_distances = distance_lower_left + distance_lower_right +
441  distance_upper_left + distance_upper_right;
442  distance_lower_left /= sum_distances;
443  distance_lower_right /= sum_distances;
444  distance_upper_left /= sum_distances;
445  distance_upper_right /= sum_distances;
446 
447  // set color depending on distance from next 4 texels
448  // (we map the data from std::array<uint8_t, 3> to an Eigen::Map, then cast that
449  // to float to multiply with the float-scalar distance.)
450  // (this is untested!)
451  const Vector3f color_upper_left =
452  Eigen::Map<const Eigen::Matrix<std::uint8_t, 1, 3>>(
453  image(floor(src_texel[1]), floor(src_texel[0])).data(), 3)
454  .cast<float>() *
455  distance_upper_left;
456  const Vector3f color_upper_right =
457  Eigen::Map<const Eigen::Matrix<std::uint8_t, 1, 3>>(
458  image(floor(src_texel[1]), ceil(src_texel[0])).data(), 3)
459  .cast<float>() *
460  distance_upper_right;
461  const Vector3f color_lower_left =
462  Eigen::Map<const Eigen::Matrix<std::uint8_t, 1, 3>>(
463  image(ceil(src_texel[1]), floor(src_texel[0])).data(), 3)
464  .cast<float>() *
465  distance_lower_left;
466  const Vector3f color_lower_right =
467  Eigen::Map<const Eigen::Matrix<std::uint8_t, 1, 3>>(
468  image(ceil(src_texel[1]), ceil(src_texel[0])).data(), 3)
469  .cast<float>() *
470  distance_lower_right;
471 
472  // isomap(y, x)[color] = color_upper_left + color_upper_right + color_lower_left +
473  // color_lower_right;
474  isomap(y, x)[0] = static_cast<std::uint8_t>(
475  glm::clamp(color_upper_left[0] + color_upper_right[0] + color_lower_left[0] +
476  color_lower_right[0],
477  0.f, 255.0f));
478  isomap(y, x)[1] = static_cast<std::uint8_t>(
479  glm::clamp(color_upper_left[1] + color_upper_right[1] + color_lower_left[1] +
480  color_lower_right[1],
481  0.f, 255.0f));
482  isomap(y, x)[2] = static_cast<std::uint8_t>(
483  glm::clamp(color_upper_left[2] + color_upper_right[2] + color_lower_left[2] +
484  color_lower_right[2],
485  0.f, 255.0f));
486  isomap(y, x)[3] = static_cast<std::uint8_t>(alpha_value); // pixel is visible
487  }
488  // NearestNeighbour mapping: set color of pixel to color of nearest texel
489  else if (mapping_type == TextureInterpolation::NearestNeighbour)
490  {
491 
492  // calculate corresponding position of dst_coord pixel center in image (src)
493  const Vector3f homogenous_dst_coord(x, y, 1.0f);
494  const Vector2f src_texel = warp_mat_org_inv * homogenous_dst_coord;
495 
496  if ((round(src_texel[1]) < image.rows) && (round(src_texel[0]) < image.cols) &&
497  round(src_texel[0]) > 0 && round(src_texel[1]) > 0)
498  {
499  isomap(y, x)[0] = image(round(src_texel[1]), round(src_texel[0]))[0];
500  isomap(y, x)[1] = image(round(src_texel[1]), round(src_texel[0]))[1];
501  isomap(y, x)[2] = image(round(src_texel[1]), round(src_texel[0]))[2];
502  isomap(y, x)[3] = static_cast<std::uint8_t>(alpha_value); // pixel is visible
503  }
504  }
505  }
506  }
507  }
508  }; // end lambda auto extract_triangle();
509  results.emplace_back(std::async(extract_triangle));
510  } // end for all mesh.tvi
511  // Collect all the launched tasks:
512  for (auto&& r : results)
513  {
514  r.get();
515  }
516 
517  // Workaround for the black line in the isomap (see GitHub issue #4):
518  /*if (mesh.texcoords.size() <= 3448)
519  {
520  isomap = detail::interpolate_black_line(isomap);
521  } */
522 
523  return isomap;
524 };

669 // Workaround for the pixels that don't get filled in extract_texture().
670 // There's a vertical line of missing values in the middle of the isomap,
671 // as well as a few pixels on a horizontal line around the mouth. They
672 // manifest themselves as black lines in the final isomap. This function
673 // just fills these missing values by interpolating between two neighbouring
674 // pixels. See GitHub issue #4.
675 inline core::Image4u interpolate_black_line(core::Image4u& isomap)
676 {
677  // Replace the vertical black line ("missing data"):
678  using RGBAType = Eigen::Matrix<std::uint8_t, 1, 4>;
679  using Eigen::Map;
680  const int col = isomap.cols / 2;
681  for (int row = 0; row < isomap.rows; ++row)
682  {
683  if (isomap(row, col) == std::array<std::uint8_t, 4>{0, 0, 0, 0})
684  {
685  Eigen::Vector4f pixel_val =
686  Map<const RGBAType>(isomap(row, col - 1).data(), 4).cast<float>() * 0.5f +
687  Map<const RGBAType>(isomap(row, col + 1).data(), 4).cast<float>() * 0.5f;
688  isomap(row,
689  col) = {static_cast<std::uint8_t>(pixel_val[0]), static_cast<std::uint8_t>(pixel_val[1]),
690  static_cast<std::uint8_t>(pixel_val[2]), static_cast<std::uint8_t>(pixel_val[3])};
691  }
692  }
693 
694  // Replace the horizontal line around the mouth that occurs in the
695  // isomaps of resolution 512x512 and higher:
696  if (isomap.rows == 512) // num cols is 512 as well
697  {
698  const int r = 362;
699  for (int c = 206; c <= 306; ++c)
700  {
701  if (isomap(r, c) == std::array<std::uint8_t, 4>{0, 0, 0, 0})
702  {
703  Eigen::Vector4f pixel_val =
704  Map<const RGBAType>(isomap(r - 1, c).data(), 4).cast<float>() * 0.5f +
705  Map<const RGBAType>(isomap(r + 1, c).data(), 4).cast<float>() * 0.5f;
706  isomap(r, c) = {
707  static_cast<std::uint8_t>(pixel_val[0]), static_cast<std::uint8_t>(pixel_val[1]),
708  static_cast<std::uint8_t>(pixel_val[2]), static_cast<std::uint8_t>(pixel_val[3])};
709  }
710  }
711  }
