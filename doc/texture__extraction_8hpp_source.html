140  // Note: If there's a performance problem, there's no need to capture the whole mesh - we could capture only the three required vertices with their texcoords.
141  auto extract_triangle = [&mesh, &affine_camera_matrix, &triangle_indices, &depthbuffer, &isomap, &mapping_type, &image, &compute_view_angle]() {
142 
143  // Find out if the current triangle is visible:
144  // We do a second rendering-pass here. We use the depth-buffer of the final image, and then, here,
145  // check if each pixel in a triangle is visible. If the whole triangle is visible, we use it to extract
146  // the texture.
147  // Possible improvement: - If only part of the triangle is visible, split it
148 
149  // This could be optimized in 2 ways though:
150  // - Use render(), or as in render(...), transfer the vertices once, not in a loop over all triangles (vertices are getting transformed multiple times)
151  // - We transform them later (below) a second time. Only do it once.
152 
153  cv::Vec4f v0_as_Vec4f(mesh.vertices[triangle_indices[0]].x, mesh.vertices[triangle_indices[0]].y, mesh.vertices[triangle_indices[0]].z, mesh.vertices[triangle_indices[0]].w);
154  cv::Vec4f v1_as_Vec4f(mesh.vertices[triangle_indices[1]].x, mesh.vertices[triangle_indices[1]].y, mesh.vertices[triangle_indices[1]].z, mesh.vertices[triangle_indices[1]].w);
155  cv::Vec4f v2_as_Vec4f(mesh.vertices[triangle_indices[2]].x, mesh.vertices[triangle_indices[2]].y, mesh.vertices[triangle_indices[2]].z, mesh.vertices[triangle_indices[2]].w);
156 
157  // Project the triangle vertices to screen coordinates, and use the depthbuffer to check whether the triangle is visible:
158  const Vec4f v0 = Mat(affine_camera_matrix * Mat(v0_as_Vec4f));
159  const Vec4f v1 = Mat(affine_camera_matrix * Mat(v1_as_Vec4f));
160  const Vec4f v2 = Mat(affine_camera_matrix * Mat(v2_as_Vec4f));
161 
162  if (!detail::is_triangle_visible(glm::tvec4<float>(v0[0], v0[1], v0[2], v0[3]), glm::tvec4<float>(v1[0], v1[1], v1[2], v1[3]), glm::tvec4<float>(v2[0], v2[1], v2[2], v2[3]), depthbuffer))
163  {
164  //continue;
165  return;
166  }
167 
168  float alpha_value;
169  if (compute_view_angle)
170  {
171  // Calculate how well visible the current triangle is:
172  // (in essence, the dot product of the viewing direction (0, 0, 1) and the face normal)
173  const Vec3f face_normal = calculate_face_normal(Vec3f(Mat(v0_as_Vec4f).rowRange(0, 3)), Vec3f(Mat(v1_as_Vec4f).rowRange(0, 3)), Vec3f(Mat(v2_as_Vec4f).rowRange(0, 3)));
174  // Transform the normal to "screen" (kind of "eye") space using the upper 3x3 part of the affine camera matrix (=the translation can be ignored):
175  Vec3f face_normal_transformed = Mat(affine_camera_matrix.rowRange(0, 3).colRange(0, 3) * Mat(face_normal));
176  face_normal_transformed /= cv::norm(face_normal_transformed, cv::NORM_L2); // normalise to unit length
177  // Implementation notes regarding the affine camera matrix and the sign:
178  // If the matrix given were the model_view matrix, the sign would be correct.
179  // However, affine_camera_matrix includes glm::ortho, which includes a z-flip.
180  // So we need to flip one of the two signs.
181  // * viewing_direction(0.0f, 0.0f, 1.0f) is correct if affine_camera_matrix were only a model_view matrix
182  // * affine_camera_matrix includes glm::ortho, which flips z, so we flip the sign of viewing_direction.
183  // We don't need the dot product since viewing_direction.xy are 0 and .z is 1:
184  const float angle = -face_normal_transformed[2]; // flip sign, see above
185  assert(angle >= -1.f && angle <= 1.f);
186  // angle is [-1, 1].
187  // * +1 means 0° (same direction)
188  // * 0 means 90°
189  // * -1 means 180° (facing opposite directions)
190  // It's a linear relation, so +0.5 is 45° etc.
191  // An angle larger than 90° means the vertex won't be rendered anyway (because it's back-facing) so we encode 0° to 90°.
192  if (angle < 0.0f) {
193  alpha_value = 0.0f;
194  } else {
195  alpha_value = angle * 255.0f;
196  }
197  }
198  else {
199  // no visibility angle computation - if the triangle/pixel is visible, set the alpha chan to 255 (fully visible pixel).
200  alpha_value = 255.0f;
201  }
202 
203  // Todo: Documentation
204  cv::Point2f src_tri[3];
205  cv::Point2f dst_tri[3];
206 
207  Vec4f vec(mesh.vertices[triangle_indices[0]][0], mesh.vertices[triangle_indices[0]][1], mesh.vertices[triangle_indices[0]][2], 1.0f);
208  Vec4f res = Mat(affine_camera_matrix * Mat(vec));
209  src_tri[0] = Vec2f(res[0], res[1]);
210 
211  vec = Vec4f(mesh.vertices[triangle_indices[1]][0], mesh.vertices[triangle_indices[1]][1], mesh.vertices[triangle_indices[1]][2], 1.0f);
212  res = Mat(affine_camera_matrix * Mat(vec));
213  src_tri[1] = Vec2f(res[0], res[1]);
214 
215  vec = Vec4f(mesh.vertices[triangle_indices[2]][0], mesh.vertices[triangle_indices[2]][1], mesh.vertices[triangle_indices[2]][2], 1.0f);
216  res = Mat(affine_camera_matrix * Mat(vec));
217  src_tri[2] = Vec2f(res[0], res[1]);
218 
219  dst_tri[0] = cv::Point2f(isomap.cols*mesh.texcoords[triangle_indices[0]][0], isomap.rows*mesh.texcoords[triangle_indices[0]][1] - 1.0f);
220  dst_tri[1] = cv::Point2f(isomap.cols*mesh.texcoords[triangle_indices[1]][0], isomap.rows*mesh.texcoords[triangle_indices[1]][1] - 1.0f);
221  dst_tri[2] = cv::Point2f(isomap.cols*mesh.texcoords[triangle_indices[2]][0], isomap.rows*mesh.texcoords[triangle_indices[2]][1] - 1.0f);
222 
223  // We now have the source triangles in the image and the source triangle in the isomap
224  // We use the inverse/ backward mapping approach, so we want to find the corresponding texel (texture-pixel) for each pixel in the isomap
225 
226  // Get the inverse Affine Transform from original image: from dst (pixel in isomap) to src (in image)
227  Mat warp_mat_org_inv = cv::getAffineTransform(dst_tri, src_tri);
228  warp_mat_org_inv.convertTo(warp_mat_org_inv, CV_32FC1);
229 
230  // We now loop over all pixels in the triangle and select, depending on the mapping type, the corresponding texel(s) in the source image
231  for (int x = min(dst_tri[0].x, min(dst_tri[1].x, dst_tri[2].x)); x < max(dst_tri[0].x, max(dst_tri[1].x, dst_tri[2].x)); ++x) {
232  for (int y = min(dst_tri[0].y, min(dst_tri[1].y, dst_tri[2].y)); y < max(dst_tri[0].y, max(dst_tri[1].y, dst_tri[2].y)); ++y) {
233  if (detail::is_point_in_triangle(cv::Point2f(x, y), dst_tri[0], dst_tri[1], dst_tri[2])) {
234 
235  // As the coordinates of the transformed pixel in the image will most likely not lie on a texel, we have to choose how to
236  // calculate the pixel colors depending on the next texels
237  // there are three different texture interpolation methods: area, bilinear and nearest neighbour
238 
239  // Area mapping: calculate mean color of texels in transformed pixel area
240  if (mapping_type == TextureInterpolation::Area) {
241 
242  // calculate positions of 4 corners of pixel in image (src)
243  Vec3f homogenous_dst_upper_left(x - 0.5f, y - 0.5f, 1.0f);
244  Vec3f homogenous_dst_upper_right(x + 0.5f, y - 0.5f, 1.0f);
245  Vec3f homogenous_dst_lower_left(x - 0.5f, y + 0.5f, 1.0f);
246  Vec3f homogenous_dst_lower_right(x + 0.5f, y + 0.5f, 1.0f);
247 
248  Vec2f src_texel_upper_left = Mat(warp_mat_org_inv * Mat(homogenous_dst_upper_left));
249  Vec2f src_texel_upper_right = Mat(warp_mat_org_inv * Mat(homogenous_dst_upper_right));
250  Vec2f src_texel_lower_left = Mat(warp_mat_org_inv * Mat(homogenous_dst_lower_left));
251  Vec2f src_texel_lower_right = Mat(warp_mat_org_inv * Mat(homogenous_dst_lower_right));
252 
253  float min_a = min(min(src_texel_upper_left[0], src_texel_upper_right[0]), min(src_texel_lower_left[0], src_texel_lower_right[0]));
254  float max_a = max(max(src_texel_upper_left[0], src_texel_upper_right[0]), max(src_texel_lower_left[0], src_texel_lower_right[0]));
255  float min_b = min(min(src_texel_upper_left[1], src_texel_upper_right[1]), min(src_texel_lower_left[1], src_texel_lower_right[1]));
256  float max_b = max(max(src_texel_upper_left[1], src_texel_upper_right[1]), max(src_texel_lower_left[1], src_texel_lower_right[1]));
257 
258  cv::Vec3i color;
259  int num_texels = 0;
260 
261  // loop over square in which quadrangle out of the four corners of pixel is
262  for (int a = ceil(min_a); a <= floor(max_a); ++a)
263  {
264  for (int b = ceil(min_b); b <= floor(max_b); ++b)
265  {
266  // check if texel is in quadrangle
267  if (detail::is_point_in_triangle(cv::Point2f(a, b), src_texel_upper_left, src_texel_lower_left, src_texel_upper_right) || detail::is_point_in_triangle(cv::Point2f(a, b), src_texel_lower_left, src_texel_upper_right, src_texel_lower_right)) {
268  if (a < image.cols && b < image.rows) { // check if texel is in image
269  num_texels++;
270  color += image.at<Vec3b>(b, a);
271  }
272  }
273  }
274  }
275  if (num_texels > 0)
276  color = color / num_texels;
277  else { // if no corresponding texel found, nearest neighbour interpolation
278  // calculate corresponding position of dst_coord pixel center in image (src)
279  Vec3f homogenous_dst_coord = Vec3f(x, y, 1.0f);
280  Vec2f src_texel = Mat(warp_mat_org_inv * Mat(homogenous_dst_coord));
281 
282  if ((cvRound(src_texel[1]) < image.rows) && cvRound(src_texel[0]) < image.cols) {
283  color = image.at<Vec3b>(cvRound(src_texel[1]), cvRound(src_texel[0]));
284  }
285  }
286  isomap.at<Vec3b>(y, x) = color;
287  }
288  // Bilinear mapping: calculate pixel color depending on the four neighbouring texels
289  else if (mapping_type == TextureInterpolation::Bilinear) {
290 
291  // calculate corresponding position of dst_coord pixel center in image (src)
292  Vec3f homogenous_dst_coord(x, y, 1.0f);
293  Vec2f src_texel = Mat(warp_mat_org_inv * Mat(homogenous_dst_coord));
294 
295  // calculate euclidean distances to next 4 texels
296  using std::sqrt;
297  using std::pow;
298  float distance_upper_left = sqrt(pow(src_texel[0] - floor(src_texel[0]), 2) + pow(src_texel[1] - floor(src_texel[1]), 2));
299  float distance_upper_right = sqrt(pow(src_texel[0] - floor(src_texel[0]), 2) + pow(src_texel[1] - ceil(src_texel[1]), 2));
300  float distance_lower_left = sqrt(pow(src_texel[0] - ceil(src_texel[0]), 2) + pow(src_texel[1] - floor(src_texel[1]), 2));
301  float distance_lower_right = sqrt(pow(src_texel[0] - ceil(src_texel[0]), 2) + pow(src_texel[1] - ceil(src_texel[1]), 2));
302 
303  // normalise distances that the sum of all distances is 1
304  float sum_distances = distance_lower_left + distance_lower_right + distance_upper_left + distance_upper_right;
305  distance_lower_left /= sum_distances;
306  distance_lower_right /= sum_distances;
307  distance_upper_left /= sum_distances;
308  distance_upper_right /= sum_distances;
309 
310  // set color depending on distance from next 4 texels
311  for (int color = 0; color < 3; ++color) {
312  float color_upper_left = image.at<Vec3b>(floor(src_texel[1]), floor(src_texel[0]))[color] * distance_upper_left;
313  float color_upper_right = image.at<Vec3b>(floor(src_texel[1]), ceil(src_texel[0]))[color] * distance_upper_right;
314  float color_lower_left = image.at<Vec3b>(ceil(src_texel[1]), floor(src_texel[0]))[color] * distance_lower_left;
315  float color_lower_right = image.at<Vec3b>(ceil(src_texel[1]), ceil(src_texel[0]))[color] * distance_lower_right;
316 
317  isomap.at<Vec3b>(y, x)[color] = color_upper_left + color_upper_right + color_lower_left + color_lower_right;
318  }
319  }
320  // NearestNeighbour mapping: set color of pixel to color of nearest texel
321  else if (mapping_type == TextureInterpolation::NearestNeighbour) {
322 
323  // calculate corresponding position of dst_coord pixel center in image (src)
324  const Mat homogenous_dst_coord(Vec3f(x, y, 1.0f));
325  const Vec2f src_texel = Mat(warp_mat_org_inv * homogenous_dst_coord);
326 
327  if ((cvRound(src_texel[1]) < image.rows) && (cvRound(src_texel[0]) < image.cols) && cvRound(src_texel[0]) > 0 && cvRound(src_texel[1]) > 0)
328  {
329  cv::Vec4b isomap_pixel;
330  isomap.at<cv::Vec4b>(y, x)[0] = image.at<Vec3b>(cvRound(src_texel[1]), cvRound(src_texel[0]))[0];
331  isomap.at<cv::Vec4b>(y, x)[1] = image.at<Vec3b>(cvRound(src_texel[1]), cvRound(src_texel[0]))[1];
332  isomap.at<cv::Vec4b>(y, x)[2] = image.at<Vec3b>(cvRound(src_texel[1]), cvRound(src_texel[0]))[2];
333  isomap.at<cv::Vec4b>(y, x)[3] = static_cast<uchar>(alpha_value); // pixel is visible
334  }
335  }
336  }
337  }
338  }
339  }; // end lambda auto extract_triangle();
340  results.emplace_back(std::async(extract_triangle));
341  } // end for all mesh.tvi
342  // Collect all the launched tasks:
343  for (auto&& r : results) {
344  r.get();
345  }
346 
347  // Workaround for the black line in the isomap (see GitHub issue #4):
348  if (mesh.texcoords.size() <= 3448)
349  {
350  isomap = detail::interpolate_black_line(isomap);
351  }
352 
353  return isomap;

394  // apply it?
395  // (If so, then we can change the two input matrices to this function to one (mvp_matrix)).
396  std::for_each(std::begin(mesh.vertices), std::end(mesh.vertices),
397  [&rotated_vertices, &view_model_matrix](auto&& v) {
398  rotated_vertices.push_back(view_model_matrix * v);
399  });
400  // This code is duplicated from the edge-fitting. I think I can put this into a function in the library.
401  for (const auto& vertex : rotated_vertices)
402  {
403  bool visible = true;
404  // For every tri of the rotated mesh:
405  for (auto&& tri : mesh.tvi)
406  {
407  auto& v0 = rotated_vertices[tri[0]]; // const?
408  auto& v1 = rotated_vertices[tri[1]];
409  auto& v2 = rotated_vertices[tri[2]];
410

412  vec3 ray_direction(0.0f, 0.0f, 1.0f); // we shoot the ray from the vertex towards the camera
413  auto intersect = fitting::ray_triangle_intersect(ray_origin, ray_direction, vec3(v0), vec3(v1),
414  vec3(v2), false);
415  // first is bool intersect, second is the distance t
416  if (intersect.first == true)
417  {
418  // We've hit a triangle. Ray hit its own triangle. If it's behind the ray origin, ignore the
419  // intersection:
420  // Check if in front or behind?
421  if (intersect.second.get() <= 1e-4)
422  {
423  continue; // the intersection is behind the vertex, we don't care about it
424  }
425  // Otherwise, we've hit a genuine triangle, and the vertex is not visible:
426  visible = false;
427  break;
428  }
429  }
430  visibility_ray.push_back(visible);
431  }

451  visibility_ray[tvi[2]]) // can also try using ||, but...

454  // definitely need to correct this. Probably here.

459  vec3(/* empty */),
460  vec2(

462  /* maybe 1 - ... ? */ wnd_coords[tvi[0]].y /

468  vec3(/* empty */),

494 // Workaround for the pixels that don't get filled in extract_texture().
495 // There's a vertical line of missing values in the middle of the isomap,
496 // as well as a few pixels on a horizontal line around the mouth. They
497 // manifest themselves as black lines in the final isomap. This function
498 // just fills these missing values by interpolating between two neighbouring
499 // pixels. See GitHub issue #4.
500 inline cv::Mat interpolate_black_line(cv::Mat isomap)
501 {
502  assert(isomap.type() == CV_8UC4);
503  // Replace the vertical black line ("missing data"):
504  int col = isomap.cols / 2;
505  for (int row = 0; row < isomap.rows; ++row)
506  {
507  if (isomap.at<cv::Vec4b>(row, col) == cv::Vec4b(0, 0, 0, 0))
508  {
509  isomap.at<cv::Vec4b>(row, col) = isomap.at<cv::Vec4b>(row, col - 1) * 0.5f + isomap.at<cv::Vec4b>(row, col + 1) * 0.5f;
510  }
511  }
512 
513  // Replace the horizontal line around the mouth that occurs in the
514  // isomaps of resolution 512x512 and higher:
515  if (isomap.rows == 512) // num cols is 512 as well
516  {
517  int r = 362;
518  for (int c = 206; c <= 306; ++c)
519  {
520  if (isomap.at<cv::Vec4b>(r, c) == cv::Vec4b(0, 0, 0, 0))
521  {
522  isomap.at<cv::Vec4b>(r, c) = isomap.at<cv::Vec4b>(r - 1, c) * 0.5f + isomap.at<cv::Vec4b>(r + 1, c) * 0.5f;
523  }
524  }
525  }
526  if (isomap.rows == 1024) // num cols is 1024 as well
527  {
528  int r = 724;
529  for (int c = 437; c <= 587; ++c)
530  {
531  if (isomap.at<cv::Vec4b>(r, c) == cv::Vec4b(0, 0, 0, 0))
532  {
533  isomap.at<cv::Vec4b>(r, c) = isomap.at<cv::Vec4b>(r - 1, c) * 0.5f + isomap.at<cv::Vec4b>(r + 1, c) * 0.5f;
534  }
535  }
536  r = 725;
537  for (int c = 411; c <= 613; ++c)
538  {
539  if (isomap.at<cv::Vec4b>(r, c) == cv::Vec4b(0, 0, 0, 0))
540  {
541  isomap.at<cv::Vec4b>(r, c) = isomap.at<cv::Vec4b>(r - 1, c) * 0.5f + isomap.at<cv::Vec4b>(r + 1, c) * 0.5f;
542  }
543  }
544  }
545  // Higher resolutions are probably affected as well but not used so far in practice.
546 
547  return isomap;
