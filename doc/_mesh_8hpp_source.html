109 inline void write_textured_obj(Mesh mesh, std::string filename)
110 {
111  assert((mesh.vertices.size() == mesh.colors.size() || mesh.colors.empty()) && !mesh.texcoords.empty());
112 
113  std::ofstream obj_file(filename);

116  mtl_filename.replace_extension(".mtl");
117 
118  obj_file << "mtllib " << mtl_filename.filename().string() << std::endl; // first line of the obj file
119 
120  // same as in write_obj():
121  if (mesh.colors.empty()) {
122  for (std::size_t i = 0; i < mesh.vertices.size(); ++i) {
123  obj_file << "v " << mesh.vertices[i][0] << " " << mesh.vertices[i][1] << " " << mesh.vertices[i][2] << " " << std::endl;
124  }
125  }
126  else {
127  for (std::size_t i = 0; i < mesh.vertices.size(); ++i) {
128  obj_file << "v " << mesh.vertices[i][0] << " " << mesh.vertices[i][1] << " " << mesh.vertices[i][2] << " " << mesh.colors[i][0] << " " << mesh.colors[i][1] << " " << mesh.colors[i][2] << " " << std::endl;
129  }
130  }
131  // end
132 
133  for (std::size_t i = 0; i < mesh.texcoords.size(); ++i) {
134  obj_file << "vt " << mesh.texcoords[i][0] << " " << 1.0f - mesh.texcoords[i][1] << std::endl;
135  // We invert y because Meshlab's uv origin (0, 0) is on the bottom-left
136  }
137 
138  obj_file << "usemtl FaceTexture" << std::endl; // the name of our texture (material) will be 'FaceTexture'
139 
140  for (auto&& v : mesh.tvi) {
141  // This assumes mesh.texcoords.size() == mesh.vertices.size(). The texture indices could theoretically be different (for example in the cube-mapped 3D scan)
142  // Add one because obj starts counting triangle indices at 1
143  obj_file << "f " << v[0] + 1 << "/" << v[0] + 1 << " " << v[1] + 1 << "/" << v[1] + 1 << " " << v[2] + 1 << "/" << v[2] + 1 << std::endl;
144  }
145

147  boost::filesystem::path texture_filename(filename);

149
