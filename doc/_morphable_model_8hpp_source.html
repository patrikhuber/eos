163 core::Mesh get_mean() const
164 {
165 assert(shape_model.get_data_dimension() == color_model.get_data_dimension() ||
166 !has_color_model()); // The number of vertices (= model.getDataDimension() / 3) has to be equal
167 // for both models, or, alternatively, it has to be a shape-only model.
168
169 Eigen::VectorXf shape = shape_model.get_mean();
170 const Eigen::VectorXf color = color_model.get_mean();
171
172 // If there is a PCA expression model, add that model's mean. If there is no expression model, or
173 // blendshapes, there's nothing to add.
174 if (get_expression_model_type() == ExpressionModelType::PcaModel)
175 {
176 shape += cpp17::get<PcaModel>(expression_model.value()).get_mean();
177 }
178
179 core::Mesh mesh;
180 if (has_texture_coordinates())
181 {
182 mesh =
183 sample_to_mesh(shape, color, shape_model.get_triangle_list(), color_model.get_triangle_list(),
184 texture_coordinates, texture_triangle_indices);
185 } else
186 {
187 mesh = sample_to_mesh(shape, color, shape_model.get_triangle_list(),
188 color_model.get_triangle_list());
189 }
190 return mesh;
191 };

297 core::Mesh draw_sample(std::vector<float> shape_coefficients, std::vector<float> expression_coefficients,
298 std::vector<float> color_coefficients) const
299 {
300 assert(shape_model.get_data_dimension() == color_model.get_data_dimension() ||
301 !has_color_model()); // The number of vertices (= model.getDataDimension() / 3) has to be equal
302 // for both models, or, alternatively, it has to be a shape-only model.
303 assert(has_separate_expression_model());
304
305 Eigen::VectorXf shape_sample;
306 Eigen::VectorXf expression_sample;
307 Eigen::VectorXf color_sample;
308
309 if (shape_coefficients.empty())
310 {
311 shape_sample = shape_model.get_mean();
312 } else
313 {
314 shape_sample = shape_model.draw_sample(shape_coefficients);
315 }
316 // Get a sample of the expression model, depending on whether it's a PcaModel or Blendshapes:
317 if (get_expression_model_type() == ExpressionModelType::PcaModel)
318 {
319 const auto& pca_expression_model = cpp17::get<PcaModel>(expression_model.value());
320 assert(pca_expression_model.get_data_dimension() == shape_model.get_data_dimension());
321 if (expression_coefficients.empty())
322 {
323 expression_sample = pca_expression_model.get_mean();
324 } else
325 {
326 expression_sample = pca_expression_model.draw_sample(expression_coefficients);
327 }
328 } else if (get_expression_model_type() == ExpressionModelType::Blendshapes)
329 {
330 const auto& expression_blendshapes = cpp17::get<Blendshapes>(expression_model.value());
331 assert(expression_blendshapes.size() > 0);
332 assert(expression_blendshapes[0].deformation.rows() == shape_model.get_data_dimension());
333 if (expression_coefficients.empty())
334 {
335 expression_sample.setZero(expression_blendshapes[0].deformation.rows());
336 } else
337 {
338 expression_sample = to_matrix(expression_blendshapes) *
339 Eigen::Map<const Eigen::VectorXf>(expression_coefficients.data(),
340 expression_coefficients.size());
341 }
342 } else
343 {
344 throw std::runtime_error("This MorphableModel doesn't contain an expression model in the form of "
345 "a PcaModel or Blendshapes.");
346 }
347 shape_sample += expression_sample;
348
349 if (color_coefficients.empty())
350 {
351 color_sample = color_model.get_mean();
352 } else
353 {
354 color_sample = color_model.draw_sample(color_coefficients);
355 }
356
357 core::Mesh mesh;
358 if (has_texture_coordinates())
359 {
360 mesh = sample_to_mesh(shape_sample, color_sample, shape_model.get_triangle_list(),
361 color_model.get_triangle_list(), texture_coordinates,
362 texture_triangle_indices);
363 } else
364 {
365 mesh = sample_to_mesh(shape_sample, color_sample, shape_model.get_triangle_list(),
366 color_model.get_triangle_list());
367 }
368 return mesh;
369 };

441 ExpressionModelType get_expression_model_type() const
442 {
443 if (!expression_model)
444 {
445 return ExpressionModelType::None;
446 }
447 if (cpp17::holds_alternative<Blendshapes>(expression_model.value()))
448 {
449 return ExpressionModelType::Blendshapes;
450 } else if (cpp17::holds_alternative<PcaModel>(expression_model.value()))
451 {
452 return ExpressionModelType::PcaModel;
453 } else
454 {
455 // We should never get here - but this silences the "not all control paths return a value"
456 // compiler warning.
457 throw std::runtime_error("The ExpressionModel contains something, but it's not Blendshapes or a "
458 "PcaModel. This should not happen.");
459 }

485 friend class cereal::access;
494 template <class Archive>
495 void serialize(Archive& archive, const std::uint32_t version)
496 {
497 if (version < 1)
498 {
499 throw std::runtime_error("The model file you are trying to load is in an old format. Please "
500 "download the most recent model files.");
501 }
502 if (version == 1)
503 {
504 archive(CEREAL_NVP(shape_model), CEREAL_NVP(color_model), CEREAL_NVP(texture_coordinates));
505 } else if (version == 2)
506 {
507 archive(CEREAL_NVP(shape_model), CEREAL_NVP(color_model), CEREAL_NVP(expression_model),
508 CEREAL_NVP(texture_coordinates));
509 } else if (version == 3)
510 {
511 archive(CEREAL_NVP(shape_model), CEREAL_NVP(color_model), CEREAL_NVP(expression_model),
512 CEREAL_NVP(landmark_definitions), CEREAL_NVP(texture_coordinates));
513 } else if (version == 4)
514 {
515 archive(CEREAL_NVP(shape_model), CEREAL_NVP(color_model), CEREAL_NVP(expression_model),
516 CEREAL_NVP(landmark_definitions), CEREAL_NVP(texture_coordinates),
517 CEREAL_NVP(texture_triangle_indices));
518 } else
519 {
520 throw std::runtime_error("The model file you are trying to load has an unknown version number: " +
521 std::to_string(version));
522 }

584inline core::Mesh sample_to_mesh(
585 const Eigen::VectorXf& shape_instance, const Eigen::VectorXf& color_instance,
586 const std::vector<std::array<int, 3>>& tvi, const std::vector<std::array<int, 3>>& tci,
587 const std::vector<std::array<double, 2>>&
588 texture_coordinates /* = std::vector<std::array<double, 2>>() */,
589 const std::vector<std::array<int, 3>>& texture_triangle_indices /* = std::vector<std::array<int, 3>>() */)
590{
591 assert(shape_instance.rows() == color_instance.rows() ||
592 color_instance.size() == 0); // The number of vertices (= model.getDataDimension() / 3) has to be
593 // equal for both models, or, alternatively, it has to be a shape-only
594 // model.
595 assert(texture_coordinates.size() == 0 || texture_coordinates.size() == (shape_instance.rows() / 3) ||
596 !texture_triangle_indices
597 .empty()); // No texture coordinates are ok. If there are texture
598 // coordinates given, their number needs to be identical to the
599 // number of vertices, or texture_triangle_indices needs to be given.
600
601 const auto num_vertices = shape_instance.rows() / 3;
602
603 core::Mesh mesh;
604
605 // Construct the mesh vertices:
606 mesh.vertices.resize(num_vertices);
607 for (auto i = 0; i < num_vertices; ++i)
608 {
609 mesh.vertices[i] = Eigen::Vector3f(
610 shape_instance(i * 3 + 0), shape_instance(i * 3 + 1),
611 shape_instance(i * 3 + 2)); // Note: This can probably be simplified now, Eigen on both sides!
612 }
613
614 // Assign the vertex colour information if it's not a shape-only model:
615 if (color_instance.size() > 0)
616 {
617 mesh.colors.resize(num_vertices);
618 for (auto i = 0; i < num_vertices; ++i)
619 {
620 mesh.colors[i] = Eigen::Vector3f(
621 cpp17::clamp(color_instance(i * 3 + 0), 0.0f, 1.0f),
622 cpp17::clamp(color_instance(i * 3 + 1), 0.0f, 1.0f),
623 cpp17::clamp(color_instance(i * 3 + 2), 0.0f, 1.0f)); // We use RGB order everywhere.
624 }
625 }
626
627 // Assign the triangle lists:
628 mesh.tvi = tvi;
629 mesh.tci = tci; // tci will be empty in case of a shape-only model
630
631 // Texture coordinates, if the model has them:
632 if (!texture_coordinates.empty())
633 {
634 mesh.texcoords.resize(texture_coordinates.size());
635 for (auto i = 0; i < texture_coordinates.size(); ++i)
636 {
637 mesh.texcoords[i] = Eigen::Vector2f(texture_coordinates[i][0], texture_coordinates[i][1]);
638 }
639 if (!texture_triangle_indices.empty())
640 {
641 mesh.tti = texture_triangle_indices;
642 }
643 }
644
645 return mesh;
646};
