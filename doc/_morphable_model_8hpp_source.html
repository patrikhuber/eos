297 inline core::Mesh sample_to_mesh(const Eigen::VectorXf& shape_instance, const Eigen::VectorXf& color_instance, const std::vector<std::array<int, 3>>& tvi, const std::vector<std::array<int, 3>>& tci, const std::vector<std::array<double, 2>>& texture_coordinates /* = std::vector<std::array<double, 2>>() */)
298 {
299  assert(shape_instance.rows() == color_instance.rows() || color_instance.size() == 0); // The number of vertices (= model.getDataDimension() / 3) has to be equal for both models, or, alternatively, it has to be a shape-only model.
300 
301  auto num_vertices = shape_instance.rows() / 3;
302 
303  core::Mesh mesh;
304 
305  // Construct the mesh vertices:
306  mesh.vertices.resize(num_vertices);
307  for (auto i = 0; i < num_vertices; ++i) {
308  mesh.vertices[i] = glm::tvec4<float>(shape_instance(i * 3 + 0), shape_instance(i * 3 + 1), shape_instance(i * 3 + 2), 1.0f);
309  }
310 
311  // Assign the vertex colour information if it's not a shape-only model:
312  if (color_instance.size() > 0) {
313  mesh.colors.resize(num_vertices);
314  for (auto i = 0; i < num_vertices; ++i) {
315  mesh.colors[i] = glm::tvec3<float>(color_instance(i * 3 + 0), color_instance(i * 3 + 1), color_instance(i * 3 + 2)); // We use RGB order everywhere
316  }
317  }
318 
319  // Assign the triangle lists:
320  mesh.tvi = tvi;
321  mesh.tci = tci; // tci will be empty in case of a shape-only model
322 
323  // Texture coordinates, if the model has them:
324  if (!texture_coordinates.empty()) {
325  mesh.texcoords.resize(num_vertices);
326  for (auto i = 0; i < num_vertices; ++i) {
327  mesh.texcoords[i] = glm::tvec2<float>(texture_coordinates[i][0], texture_coordinates[i][1]);
328  }
329  }
330 
331  return mesh;
