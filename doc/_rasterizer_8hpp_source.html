66 template <typename T>
67 void raster_triangle(const detail::Vertex<T>& point_a, const detail::Vertex<T>& point_b,
68 const detail::Vertex<T>& point_c, const cpp17::optional<Texture>& texture)
69 {
70 // We already calculated this in the culling/clipping stage. Maybe we should save/cache it after all.
71 const auto boundingBox = detail::calculate_clipped_bounding_box(
72 Eigen::Vector2<T>(point_a.position.x(), point_a.position.y()),
73 Eigen::Vector2<T>(point_b.position.x(), point_b.position.y()),
74 Eigen::Vector2<T>(point_c.position.x(), point_c.position.y()), viewport_width, viewport_height);
75 const auto min_x = boundingBox.x;
76 const auto max_x = boundingBox.x + boundingBox.width;
77 const auto min_y = boundingBox.y;
78 const auto max_y = boundingBox.y + boundingBox.height;
79
80 // These are triangle-specific, i.e. calculate once per triangle.
81 // These ones are needed for perspective correct lambdas! (as well as mipmapping)
82 const auto& one_over_w0 = point_a.position[3];
83 const auto& one_over_w1 = point_b.position[3];
84 const auto& one_over_w2 = point_c.position[3];
85
86 // These are triangle-specific, i.e. calculate once per triangle.
87 // For partial derivatives computation (for mipmapping, texturing) (they work on screen-space coords):
88 using eos::render::detail::plane;
89 const auto alpha_plane = plane(
90 Eigen::Vector3<T>(point_a.position[0], point_a.position[1], point_a.texcoords[0] * one_over_w0),
91 Eigen::Vector3<T>(point_b.position[0], point_b.position[1], point_b.texcoords[0] * one_over_w1),
92 Eigen::Vector3<T>(point_c.position[0], point_c.position[1], point_c.texcoords[0] * one_over_w2));
93 const auto beta_plane = plane(
94 Eigen::Vector3<T>(point_a.position[0], point_a.position[1], point_a.texcoords[1] * one_over_w0),
95 Eigen::Vector3<T>(point_b.position[0], point_b.position[1], point_b.texcoords[1] * one_over_w1),
96 Eigen::Vector3<T>(point_c.position[0], point_c.position[1], point_c.texcoords[1] * one_over_w2));
97 const auto gamma_plane =
98 plane(Eigen::Vector3<T>(point_a.position[0], point_a.position[1], one_over_w0),
99 Eigen::Vector3<T>(point_b.position[0], point_b.position[1], one_over_w1),
100 Eigen::Vector3<T>(point_c.position[0], point_c.position[1], one_over_w2));
101 const auto one_over_alpha_c = 1.0f / alpha_plane.c;
102 const auto one_over_beta_c = 1.0f / beta_plane.c;
103 const auto one_over_gamma_c = 1.0f / gamma_plane.c;
104 const auto alpha_ffx = -alpha_plane.a * one_over_alpha_c;
105 const auto beta_ffx = -beta_plane.a * one_over_beta_c;
106 const auto gamma_ffx = -gamma_plane.a * one_over_gamma_c;
107 const auto alpha_ffy = -alpha_plane.b * one_over_alpha_c;
108 const auto beta_ffy = -beta_plane.b * one_over_beta_c;
109 const auto gamma_ffy = -gamma_plane.b * one_over_gamma_c;
110
111 for (int yi = min_y; yi <= max_y; ++yi)
112 {
113 for (int xi = min_x; xi <= max_x; ++xi)
114 {
115 // we want centers of pixels to be used in computations. Todo: Do we? Do we pass it with or
116 // without +0.5 to the FragShader?
117 const float x = static_cast<float>(xi) + 0.5f; // double? T?
118 const float y = static_cast<float>(yi) + 0.5f;
119
120 // These will be used for barycentric weights computation
121 using detail::implicit_line;
122 const double one_over_v0ToLine12 =
123 1.0 / implicit_line(point_a.position[0], point_a.position[1], point_b.position,
124 point_c.position);
125 const double one_over_v1ToLine20 =
126 1.0 / implicit_line(point_b.position[0], point_b.position[1], point_c.position,
127 point_a.position);
128 const double one_over_v2ToLine01 =
129 1.0 / implicit_line(point_c.position[0], point_c.position[1], point_a.position,
130 point_b.position);
131 // Affine barycentric weights:
132 double alpha = implicit_line(x, y, point_b.position, point_c.position) * one_over_v0ToLine12;
133 double beta = implicit_line(x, y, point_c.position, point_a.position) * one_over_v1ToLine20;
134 double gamma = implicit_line(x, y, point_a.position, point_b.position) * one_over_v2ToLine01;
135
136 // if pixel (x, y) is inside the triangle or on one of its edges
137 if (alpha >= 0 && beta >= 0 && gamma >= 0)
138 {
139 const int pixel_index_row = yi;
140 const int pixel_index_col = xi;
141
142 // TODO: Check this one. What about perspective?
143 const double z_affine = alpha * static_cast<double>(point_a.position[2]) +
144 beta * static_cast<double>(point_b.position[2]) +
145 gamma * static_cast<double>(point_c.position[2]);
146
147 bool draw = true;
148 if (enable_far_clipping)
149 {
150 if (z_affine > 1.0)
151 {
152 draw = false;
153 }
154 }
155
156 bool passes_depth_test = false;
157 if (enable_depth_test)
158 {
159 // If enable_depth_test=false, avoid accessing the depthbuffer at all - it might be
160 // empty or have other dimensions.
161 passes_depth_test =
162 (z_affine < depthbuffer(pixel_index_row, pixel_index_col));
163 }
164 // The '<= 1.0' clips against the far-plane in NDC. We clip against the near-plane
165 // earlier.
166 // if (z_affine < depthbuffer.at<double>(pixelIndexRow, pixelIndexCol)/* && z_affine <=
167 // 1.0*/) // what to do in ortho case without n/f "squashing"? should we always squash? or
168 // a flag?
169 if ((passes_depth_test && draw) || enable_depth_test == false)
170 {
171 // perspective-correct barycentric weights
172 // Todo: Check this in the original/older implementation, i.e. if all is still
173 // perspective-correct. I think so. Also compare 1:1 with OpenGL.
174 // This is the default case when rendering - we perspectively correct the barycentric
175 // weights. However when extracting texture, it seems we don't want to do that, and
176 // pass the uncorrected lambda in that case. This switch allows us to do that.
177 if (perspective_correct_barycentric_weights)
178 {
179 double d = alpha * one_over_w0 + beta * one_over_w1 + gamma * one_over_w2;
180 d = 1.0 / d;
181 alpha *= d * one_over_w0; // In case of affine cam matrix, everything is 1 and
182 // a/b/g don't get changed.
183 beta *= d * one_over_w1;
184 gamma *= d * one_over_w2;
185 }
186 Eigen::Vector3<T> lambda(alpha, beta, gamma);
187
188 Eigen::Vector4<T> pixel_color;
189 if (texture)
190 {
191 // check if texture != NULL?
192 // partial derivatives (for mip-mapping, not needed for texturing otherwise!)
193 const float u_over_z =
194 -(alpha_plane.a * x + alpha_plane.b * y + alpha_plane.d) * one_over_alpha_c;
195 const float v_over_z =
196 -(beta_plane.a * x + beta_plane.b * y + beta_plane.d) * one_over_beta_c;
197 const float one_over_z =
198 -(gamma_plane.a * x + gamma_plane.b * y + gamma_plane.d) * one_over_gamma_c;
199 const float one_over_squared_one_over_z = 1.0f / std::pow(one_over_z, 2);
200
201 // partial derivatives of U/V coordinates with respect to X/Y pixel's screen
202 // coordinates
203 // These are exclusively used for the mipmap level computation (i.e. which mipmap
204 // levels to use).
205 // They're not needed for texturing otherwise at all!
206 float dudx =
207 one_over_squared_one_over_z * (alpha_ffx * one_over_z - u_over_z * gamma_ffx);
208 float dudy =
209 one_over_squared_one_over_z * (beta_ffx * one_over_z - v_over_z * gamma_ffx);
210 float dvdx =
211 one_over_squared_one_over_z * (alpha_ffy * one_over_z - u_over_z * gamma_ffy);
212 float dvdy =
213 one_over_squared_one_over_z * (beta_ffy * one_over_z - v_over_z * gamma_ffy);
214 dudx *= texture.value().mipmaps[0].width();
215 dudy *= texture.value().mipmaps[0].width();
216 dvdx *= texture.value().mipmaps[0].height();
217 dvdy *= texture.value().mipmaps[0].height();
218
219 // Why does it need x and y? Maybe some shaders (eg TexExtr?) need it?
220 pixel_color = fragment_shader.shade_triangle_pixel(
221 x, y, point_a, point_b, point_c, lambda, texture, dudx, dudy, dvdx, dvdy);
222
223 } else
224 { // We use vertex-coloring
225 // Why does it need x and y?
226 pixel_color = fragment_shader.shade_triangle_pixel(
227 x, y, point_a, point_b, point_c, lambda, texture, 0, 0, 0, 0);
228 }
229
230 // clamp bytes to 255
231 // Todo: Proper casting (rounding?)? And we don't clamp/max against 255? Use
232 // glm::clamp?
233 const unsigned char red =
234 static_cast<unsigned char>(255.0f * std::min(pixel_color[0], T(1)));
235 const unsigned char green =
236 static_cast<unsigned char>(255.0f * std::min(pixel_color[1], T(1)));
237 const unsigned char blue =
238 static_cast<unsigned char>(255.0f * std::min(pixel_color[2], T(1)));
239 const unsigned char alpha =
240 static_cast<unsigned char>(255.0f * std::min(pixel_color[3], T(1)));
241
242 // update buffers
243 colorbuffer(pixel_index_row, pixel_index_col)[0] = blue;
244 colorbuffer(pixel_index_row, pixel_index_col)[1] = green;
245 colorbuffer(pixel_index_row, pixel_index_col)[2] = red;
246 colorbuffer(pixel_index_row, pixel_index_col)[3] = alpha;
247 if (enable_depth_test) // TODO: A better name for this might be enable_zbuffer? or
248 // enable_zbuffer_test?
249 {
250 depthbuffer(pixel_index_row, pixel_index_col) = z_affine;
251 }
252 }
253 }
254 }
255 }
256 };
257
267 void clear_buffers()
268 {
269 colorbuffer = eos::core::image::constant(viewport_height, viewport_width,
270 eos::core::Pixel<std::uint8_t, 4>(255, 255, 255, 0));
271 depthbuffer =
272 eos::core::image::constant(viewport_height, viewport_width, std::numeric_limits<double>::max());
273 };
274
275private:
276 FragmentShaderType fragment_shader;
277
278public: // will eventually go private
279 bool enable_depth_test = true; // maybe get rid of this again, it was just as a hack.
280 bool perspective_correct_barycentric_weights = true;
281 bool enable_far_clipping = true;
282
283 int viewport_width;
284 int viewport_height;
285
286 eos::core::Image4u colorbuffer;
287 eos::core::Image1d depthbuffer;
288};
