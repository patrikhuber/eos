64  bool try_add(float frame_score, cv::Mat image, const fitting::FittingResult& fitting_result)
65  {
66  // Determine whether to add or not:
67  auto yaw_angle = glm::degrees(glm::yaw(fitting_result.rendering_parameters.get_rotation()));
68  auto idx = angle_to_index(yaw_angle);
69  bool add_frame = false;
70  if (bins[idx].size() < frames_per_bin) // always add when we don't have enough frames
71  add_frame =
72  true; // definitely adding - we wouldn't have to go through the for-loop on the next line.
73  for (auto&& f : bins[idx])
74  {
75  if (frame_score > f.score)
76  add_frame = true;
77  }
78  if (!add_frame)
79  {
80  return false;
81  }
82  // Add the keyframe:
83  bins[idx].push_back(video::Keyframe{frame_score, image, fitting_result});
84  if (bins[idx].size() > frames_per_bin)
85  {
86  // need to remove the lowest one:
87  std::sort(std::begin(bins[idx]), std::end(bins[idx]),
88  [](const auto& lhs, const auto& rhs) { return lhs.score > rhs.score; });
89  bins[idx].resize(frames_per_bin);
90  }
91  return true;
92  };

94  // Returns the keyframes as a vector.
95  std::vector<Keyframe> get_keyframes() const
96  {
97  std::vector<Keyframe> keyframes;
98  for (auto&& b : bins)
99  {
100  for (auto&& f : b)
101  {
102  keyframes.push_back(f);
103  }
104  }
105  return keyframes;
106  };

114  // Converts a given yaw angle to an index in the internal bins vector.
115  // Assumes 9 bins and 20° intervals.
116  static std::size_t angle_to_index(float yaw_angle)
117  {
118  if (yaw_angle <= -70.f)
119  return 0;
120  if (yaw_angle <= -50.f)
121  return 1;
122  if (yaw_angle <= -30.f)
123  return 2;
124  if (yaw_angle <= -10.f)
125  return 3;
126  if (yaw_angle <= 10.f)
127  return 4;
128  if (yaw_angle <= 30.f)
129  return 5;
130  if (yaw_angle <= 50.f)
131  return 6;
132  if (yaw_angle <= 70.f)
133  return 7;
134  return 8;
135  };

161 cv::Mat merge_weighted_mean(const std::vector<Keyframe>& keyframes,
162  const morphablemodel::MorphableModel& morphable_model,
163  const std::vector<morphablemodel::Blendshape>& blendshapes)
164 {
165  assert(keyframes.size() >= 1);
166 
167  using cv::Mat;
168  using std::vector;
169 
170  vector<Mat> isomaps;
171  for (const auto& frame_data : keyframes)
172  {
173  const Mat shape =
174  morphable_model.get_shape_model().draw_sample(frame_data.fitting_result.pca_shape_coefficients) +
175  morphablemodel::to_matrix(blendshapes) * Mat(frame_data.fitting_result.blendshape_coefficients);
176  const auto mesh =
177  morphablemodel::sample_to_mesh(shape, {}, morphable_model.get_shape_model().get_triangle_list(),
178  {}, morphable_model.get_texture_coordinates());
179  const Mat affine_camera_matrix = fitting::get_3x4_affine_camera_matrix(
180  frame_data.fitting_result.rendering_parameters, frame_data.frame.cols, frame_data.frame.rows);
181  const Mat isomap = render::extract_texture(mesh, affine_camera_matrix, frame_data.frame, true,
182  render::TextureInterpolation::NearestNeighbour, 1024);
183  isomaps.push_back(isomap);
184  }
185 
186  // Now do the actual merging:
187  Mat r = Mat::zeros(isomaps[0].rows, isomaps[0].cols, CV_32FC1);
188  Mat g = Mat::zeros(isomaps[0].rows, isomaps[0].cols, CV_32FC1);
189  Mat b = Mat::zeros(isomaps[0].rows, isomaps[0].cols, CV_32FC1);
190  Mat accumulated_weight = Mat::zeros(isomaps[0].rows, isomaps[0].cols, CV_32FC1);
191  // Currently, this just uses the weights in the alpha channel for weighting - they contain only the
192  // view-angle. We should use the keyframe's score as well. Plus the area of the source triangle.
193  for (auto&& isomap : isomaps)
194  {
195  vector<Mat> channels;
196  cv::split(isomap, channels);
197  // channels[0].convertTo(channels[0], CV_32FC1);
198  // We could avoid this explicit temporary, but then we'd have to convert both matrices
199  // to CV_32FC1 first - and manually multiply with 1/255. Not sure which one is faster.
200  // If we do it like this, the add just becomes '+=' - so I think it's fine like this.
201  // The final formula is:
202  // b += chan_0 * alpha * 1/255; (and the same for g and r respectively)
203  Mat weighted_b, weighted_g, weighted_r;
204  // // we scale the weights from [0, 255] to [0, 1]:
205  cv::multiply(channels[0], channels[3], weighted_b, 1 / 255.0, CV_32FC1);
206  cv::multiply(channels[1], channels[3], weighted_g, 1 / 255.0, CV_32FC1);
207  cv::multiply(channels[2], channels[3], weighted_r, 1 / 255.0, CV_32FC1);
208  b += weighted_b;
209  g += weighted_g;
210  r += weighted_r;
211  channels[3].convertTo(channels[3], CV_32FC1); // needed for the '/ 255.0f' below to work
212  cv::add(accumulated_weight, channels[3] / 255.0f, accumulated_weight, cv::noArray(), CV_32FC1);
213  }
214  b = b.mul(1.0 / (accumulated_weight)); // divide by number of frames used too?
215  g = g.mul(1.0 / (accumulated_weight));
216  r = r.mul(1.0 / (accumulated_weight));
217 
218  // Let's return accumulated_weight too: Normalise by num_isomaps * 255 (=maximum weight)
219  // This sets the returned weight to the average from all the isomaps. Maybe the maximum would make more
220  // sense? => Not returning anything for now.
221  // accumulated_weight = (accumulated_weight / isomaps.size()) * 255;
222 
223  Mat merged_isomap;
224  cv::merge({b, g, r}, merged_isomap);
225  merged_isomap.convertTo(merged_isomap, CV_8UC3);
226  return merged_isomap;
227 };
