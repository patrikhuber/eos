75 bool try_add(float frame_score, const ImageType& image, const fitting::FittingResult& fitting_result)
76 {
77 // Determine whether to add or not:
78 auto yaw_angle = glm::degrees(glm::yaw(fitting_result.rendering_parameters.get_rotation()));
79 auto idx = angle_to_index(yaw_angle);
80 bool add_frame = false;
81 if (bins[idx].size() < frames_per_bin) // always add when we don't have enough frames
82 add_frame =
83 true; // definitely adding - we wouldn't have to go through the for-loop on the next line.
84 for (auto&& f : bins[idx])
85 {
86 if (frame_score > f.score)
87 add_frame = true;
88 }
89 if (!add_frame)
90 {
91 return false;
92 }
93 // Add the keyframe:
94 bins[idx].push_back(video::Keyframe{frame_score, image, fitting_result});
95 if (bins[idx].size() > frames_per_bin)
96 {
97 // need to remove the lowest one:
98 std::sort(std::begin(bins[idx]), std::end(bins[idx]),
99 [](const auto& lhs, const auto& rhs) { return lhs.score > rhs.score; });
100 bins[idx].resize(frames_per_bin);
101 }
102 return true;
103 };
104
110 std::vector<Keyframe<ImageType>> get_keyframes() const
111 {
112 std::vector<Keyframe<ImageType>> keyframes;
113 for (auto&& b : bins)
114 {
115 for (auto&& f : b)
116 {
117 keyframes.push_back(f);
118 }
119 }
120 return keyframes;
121 };
122
123private:
124 using BinContent = std::vector<Keyframe<ImageType>>;
125 std::vector<BinContent> bins;
126 const int num_yaw_bins = 9;
127 int frames_per_bin;

136 static std::size_t angle_to_index(float yaw_angle)
137 {
138 if (yaw_angle <= -70.f)
139 return 0;
140 if (yaw_angle <= -50.f)
141 return 1;
142 if (yaw_angle <= -30.f)
143 return 2;
144 if (yaw_angle <= -10.f)
145 return 3;
146 if (yaw_angle <= 10.f)
147 return 4;
148 if (yaw_angle <= 30.f)
149 return 5;
150 if (yaw_angle <= 50.f)
151 return 6;
152 if (yaw_angle <= 70.f)
153 return 7;
154 return 8;
155 };
