31// Fragment shaders are a more accurate name for the same functionality as Pixel shaders. They aren't pixels
32// yet, since the output still has to past several tests (depth, alpha, stencil) as well as the fact that one
33// may be using anti-aliasing, which renders one-fragment-to-one-pixel non-true.
34// The "pixel" in "pixel shader" is a misnomer because the pixel shader doesn't operate on pixels directly.
35// The pixel shader operates on "fragments" which may or may not end up as actual pixels, depending on several
36// factors outside of the pixel shader.
37// The shaders *can not* depend on any state - they have to be able to run independently and in parallel!
38// But can they really? What about the z-test? It happens earlier at the moment - which is good?

48class VertexColoringFragmentShader
49{
50public:
64 template <typename T>
65 Eigen::Vector4<T> shade_triangle_pixel(int x, int y, const detail::Vertex<T>& point_a,
66 const detail::Vertex<T>& point_b, const detail::Vertex<T>& point_c,
67 const Eigen::Vector3<T>& lambda,
68 const cpp17::optional<Texture>& texture, float dudx, float dudy,
69 float dvdx, float dvdy)
70 {
71 // attributes interpolation
72 const Eigen::Vector3<T> color_persp =
73 lambda[0] * point_a.color + lambda[1] * point_b.color + lambda[2] * point_c.color;
74 return Eigen::Vector4<T>(color_persp[0], color_persp[1], color_persp[2], T(1));
75 };
76};

81class TexturingFragmentShader
82{
83public:
97 template <typename T>
98 Eigen::Vector4<T> shade_triangle_pixel(int x, int y, const detail::Vertex<T>& point_a,
99 const detail::Vertex<T>& point_b, const detail::Vertex<T>& point_c,
100 const Eigen::Vector3<T>& lambda,
101 const cpp17::optional<Texture>& texture, float dudx, float dudy,
102 float dvdx, float dvdy)
103 {
104 Eigen::Vector2<T> texcoords_persp =
105 lambda[0] * point_a.texcoords + lambda[1] * point_b.texcoords + lambda[2] * point_c.texcoords;
106
107 // The Texture is in BGR, thus tex2D returns BGR
108 // Todo: Think about changing that.
109 // tex2d divides the colour values by 255, so that the return value we get here is in the range [0, 1].
110 const Eigen::Vector3<T> texture_color =
111 detail::tex2d(texcoords_persp, texture.value(), dudx, dudy, dvdx, dvdy); // uses the current texture
112 const Eigen::Vector3<T> pixel_color =
113 Eigen::Vector3<T>(texture_color[2], texture_color[1], texture_color[0]);
114 // other: color.mul(tex2D(texture, texCoord));
115 return Eigen::Vector4<T>(pixel_color[0], pixel_color[1], pixel_color[2], T(1));
116 };
117};

130template <typename T>
131Eigen::Vector3<T> compute_inverse_perspectively_correct_lambda(const Eigen::Vector3<T>& lambda_world,
132 const T& one_over_w0, const T& one_over_w1,
133 const T& one_over_w2)
134{
135 const float w0 = 1 / one_over_w0;
136 const float w1 = 1 / one_over_w1;
137 const float w2 = 1 / one_over_w2;
138
139 const float d = w0 - (w0 - w1) * lambda_world.y() - (w0 - w2) * lambda_world.z();
140 if (d == 0)
141 return lambda_world;
142
143 Eigen::Vector3<T> lambda;
144
145 lambda.z() = lambda_world.z() * w2 / d;
146 lambda.y() = lambda_world.y() * w1 / d;
147
148 lambda.x() = 1 - lambda.y() - lambda.z();
149 return lambda;
150};

156class ExtractionFragmentShader
157{
158public:
172 template <typename T>
173 Eigen::Vector4<T> shade_triangle_pixel(int x, int y, const detail::Vertex<T>& point_a,
174 const detail::Vertex<T>& point_b, const detail::Vertex<T>& point_c,
175 const Eigen::Vector3<T>& lambda,
176 const cpp17::optional<Texture>& texture, float dudx, float dudy,
177 float dvdx, float dvdy)
178 {
179 const auto corrected_lambda = compute_inverse_perspectively_correct_lambda(
180 lambda, point_a.position.w(), point_b.position.w(), point_c.position.w());
181 const Eigen::Vector2<T> texcoords_persp = corrected_lambda[0] * point_a.texcoords +
182 corrected_lambda[1] * point_b.texcoords +
183 corrected_lambda[2] * point_c.texcoords;
184
185 // Texturing, no mipmapping:
186 Eigen::Vector2<T> image_tex_coords = detail::texcoord_wrap(texcoords_persp);
187 image_tex_coords[0] *= texture->mipmaps[0].width();
188 image_tex_coords[1] *= texture->mipmaps[0].height();
189 const Eigen::Vector3<T> texture_color =
190 detail::tex2d_linear(image_tex_coords, 0, texture.value()) / 255.0f;
191 const Eigen::Vector3<T> pixel_color =
192 Eigen::Vector3<T>(texture_color[2], texture_color[1], texture_color[0]);
193 return Eigen::Vector4<T>(pixel_color[0], pixel_color[1], pixel_color[2], T(1));
194 };
195};
