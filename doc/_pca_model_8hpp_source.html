149 template <class RNG>
150 Eigen::VectorXf draw_sample(RNG& engine, float sigma = 1.0f) const
151 {
152 std::normal_distribution<float> distribution(0.0f, sigma); // this constructor takes the stddev
153
154 std::vector<float> alphas(get_num_principal_components());
155
156 for (auto&& a : alphas)
157 {
158 a = distribution(engine);
159 }
160
161 return draw_sample(alphas);
162 };

172 Eigen::VectorXf draw_sample(std::vector<float> coefficients) const
173 {
174 // Fill the rest with zeros if not all coefficients are given:
175 if (coefficients.size() < get_num_principal_components())
176 {
177 coefficients.resize(get_num_principal_components());
178 }
179 const Eigen::Map<Eigen::VectorXf> alphas(coefficients.data(), coefficients.size());
180
181 const Eigen::VectorXf model_sample = mean + rescaled_pca_basis * alphas;
182
183 return model_sample;
184 };

289 std::vector<float> project(const Eigen::VectorXf& instance) const
290 {
291 // We are not transposing the basis - it's 3*num_vtx x num_components, which is correct for the
292 // projection.
293 // instance and mean are col-vectors, we transpose that to be a row-vector.
294 Eigen::VectorXf coeffs = (instance - mean).transpose() * rescaled_pca_basis;
295
296 // In the simple example below:
297 // sample = m.get_shape_model().draw_sample([0.5, -0.5, 0, 1, -1])
298 // c = m.get_shape_model().project(sample)
299 // we seem to need to divide the resulting coeffs by the eigenvalues, to make
300 // the output correct (i.e. the same as the input). I am not sure why. We should
301 // go through it on paper again. At the same time we can check github.com/patrikhuber/eos/issues/257.
302 coeffs = coeffs.array() / eigenvalues.array();
303
304 std::vector<float> coefficients(coeffs.data(), coeffs.data() + coeffs.size());
305 return coefficients;
306 };

394inline Eigen::MatrixXf rescale_pca_basis(const Eigen::MatrixXf& orthonormal_basis,
395 const Eigen::VectorXf& eigenvalues)
396{
397 Eigen::MatrixXf rescaled_basis(orthonormal_basis.rows(),
398 orthonormal_basis.cols()); // empty matrix with the same dimensions
399 const Eigen::VectorXf sqrt_of_eigenvalues = eigenvalues.array().sqrt();
400 // Rescale the basis: We multiply each eigenvector (i.e. each column) with the square root of its
401 // corresponding eigenvalue
402 for (int basis = 0; basis < orthonormal_basis.cols(); ++basis)
403 {
404 rescaled_basis.col(basis) = orthonormal_basis.col(basis) * sqrt_of_eigenvalues(basis);
405 }
406
407 return rescaled_basis;
408};
409
419inline Eigen::MatrixXf normalise_pca_basis(const Eigen::MatrixXf& rescaled_basis,
420 const Eigen::VectorXf& eigenvalues)
421{
422 Eigen::MatrixXf orthonormal_basis(rescaled_basis.rows(),
423 rescaled_basis.cols()); // empty matrix with the same dimensions
424 const Eigen::VectorXf one_over_sqrt_of_eigenvalues = eigenvalues.array().rsqrt();
425 // "Normalise" the basis: We multiply each eigenvector (i.e. each column) with 1 over the square root of
426 // its corresponding eigenvalue
427 for (int basis = 0; basis < rescaled_basis.cols(); ++basis)
428 {
429 orthonormal_basis.col(basis) = rescaled_basis.col(basis) * one_over_sqrt_of_eigenvalues(basis);
430 }
431
432 return orthonormal_basis;
433};
