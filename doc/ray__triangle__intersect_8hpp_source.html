57 inline std::pair<bool, cpp17::optional<float>>
58 ray_triangle_intersect(const glm::vec3& ray_origin, const glm::vec3& ray_direction, const glm::vec3& v0,
59  const glm::vec3& v1, const glm::vec3& v2, bool enable_backculling)
60 {
61  using glm::vec3;
62  const float epsilon = 1e-6f;
63 
64  const vec3 v0v1 = v1 - v0;
65  const vec3 v0v2 = v2 - v0;
66 
67  const vec3 pvec = glm::cross(ray_direction, v0v2);
68 
69  const float det = glm::dot(v0v1, pvec);
70  if (enable_backculling)
71  {
72  // If det is negative, the triangle is back-facing.
73  // If det is close to 0, the ray misses the triangle.
74  if (det < epsilon)
75  return {false, cpp17::nullopt};
76  } else
77  {
78  // If det is close to 0, the ray and triangle are parallel.
79  if (std::abs(det) < epsilon)
80  return {false, cpp17::nullopt};
81  }
82  const float inv_det = 1 / det;
83 
84  const vec3 tvec = ray_origin - v0;
85  const auto u = glm::dot(tvec, pvec) * inv_det;
86  if (u < 0 || u > 1)
87  return {false, cpp17::nullopt};
88 
89  const vec3 qvec = glm::cross(tvec, v0v1);
90  const auto v = glm::dot(ray_direction, qvec) * inv_det;
91  if (v < 0 || u + v > 1)
92  return {false, cpp17::nullopt};
93 
94  const auto t = glm::dot(v0v2, qvec) * inv_det;
95 
96  return {true, t};
97 };
