131  // All vertices are in clip-space now. Prepare the rasterisation stage:
132  vector<Triangle<T, P>> triangles_to_raster;
133  // This builds the (one and final) triangles to render. Meaning: The triangles formed of mesh.tvi (the
134  // ones that survived the clip/culling), plus possibly more that intersect one of the frustum planes
135  // (i.e. this can generate new triangles with new pos/vc/texcoords).
136  for (const auto& tri_indices : mesh.tvi)
137  {
138  unsigned char visibility_bits[3];
139  for (unsigned char k = 0; k < 3; k++)
140  {
141  visibility_bits[k] = 0;
142  const auto x_cc = clipspace_vertices[tri_indices[k]].x;
143  const auto y_cc = clipspace_vertices[tri_indices[k]].y;
144  const auto z_cc = clipspace_vertices[tri_indices[k]].z;
145  const auto w_cc = clipspace_vertices[tri_indices[k]].w;
146  if (x_cc < -w_cc) // true if outside of view frustum. False if on or inside the plane.
147  visibility_bits[k] |= 1; // set bit if outside of frustum
148  if (x_cc > w_cc)
149  visibility_bits[k] |= 2;
150  if (y_cc < -w_cc)
151  visibility_bits[k] |= 4;
152  if (y_cc > w_cc)
153  visibility_bits[k] |= 8;
154  if (enable_near_clipping && z_cc < -w_cc) // near plane frustum clipping
155  visibility_bits[k] |= 16;
156  if (rasterizer->enable_far_clipping && z_cc > w_cc) // far plane frustum clipping
157  visibility_bits[k] |= 32;
158  } // if all bits are 0, then it's inside the frustum
159  // all vertices are not visible - reject the triangle.
160  if ((visibility_bits[0] & visibility_bits[1] & visibility_bits[2]) > 0)
161  {
162  continue;
163  }
164  // all vertices are visible - pass the whole triangle to the rasteriser. = All bits of all 3
165  // triangles are 0.
166  if ((visibility_bits[0] | visibility_bits[1] | visibility_bits[2]) == 0)
167  {
168  // relevant part of process_prospective_tri:
169  std::array<glm::tvec4<T, P>, 3> prospective_tri{
170  divide_by_w(clipspace_vertices[tri_indices[0]]),
171  divide_by_w(clipspace_vertices[tri_indices[1]]),
172  divide_by_w(clipspace_vertices[tri_indices[2]])};
173  // We have a prospective tri in NDC coords now, with its vertices having coords [x_ndc, y_ndc,
174  // z_ndc, 1/w_clip].

176  // Replaces x and y of the NDC coords with the screen coords. Keep z and w the same.
177  const glm::tvec2<T, P> v0_screen =
178  clip_to_screen_space(prospective_tri[0].x, prospective_tri[0].y,
179  rasterizer->viewport_width, rasterizer->viewport_height);
180  prospective_tri[0].x = v0_screen.x;
181  prospective_tri[0].y = v0_screen.y;

184  rasterizer->viewport_width, rasterizer->viewport_height);

189  rasterizer->viewport_width, rasterizer->viewport_height);

191  prospective_tri[2].y = v2_screen.y;

194  // elsewhere? No?

196  {

198  glm::tvec2<T, P>(prospective_tri[0].x, prospective_tri[0].y),
199  glm::tvec2<T, P>(prospective_tri[1].x, prospective_tri[1].y),
200  glm::tvec2<T, P>(prospective_tri[2].x, prospective_tri[2].y)))
201  continue;
202  }
203 
204  // Get the bounding box of the triangle:
205  const cv::Rect boundingBox = detail::calculate_clipped_bounding_box(
206  glm::tvec2<T, P>(prospective_tri[0].x, prospective_tri[0].y),
207  glm::tvec2<T, P>(prospective_tri[1].x, prospective_tri[1].y),
208  glm::tvec2<T, P>(prospective_tri[2].x, prospective_tri[2].y), rasterizer->viewport_width,
209  rasterizer->viewport_height);
210  const auto min_x = boundingBox.x;
211  const auto max_x = boundingBox.x + boundingBox.width;
212  const auto min_y = boundingBox.y;
213  const auto max_y = boundingBox.y + boundingBox.height;
214  if (max_x <= min_x || max_y <= min_y)
215  { // Note: Can the width/height of the bbox be negative? Maybe we only need to check for
216  // equality here?
217  continue;
218  }

221  triangles_to_raster.push_back(
222  Triangle<T, P>{detail::Vertex<T, P>{prospective_tri[0], mesh.colors[tri_indices[0]],
223  mesh.texcoords[tri_indices[0]]},
224  detail::Vertex<T, P>{prospective_tri[1], mesh.colors[tri_indices[1]],
225  mesh.texcoords[tri_indices[1]]},
226  detail::Vertex<T, P>{prospective_tri[2], mesh.colors[tri_indices[2]],
227  mesh.texcoords[tri_indices[2]]}});
228  continue; // Triangle was either added or not added. Continue with next triangle.
229  }

231  // Note: It seems that this is only w.r.t. the near-plane. If a triangle is partially outside the
232  // tlbr viewport, it'll get rejected.
233  // Well, 'z' of these triangles seems to be -1, so is that really the near plane?

235  vertices.reserve(3);

238  mesh.texcoords[tri_indices[0]]});

252  // little bit unsure because Songho says the normal
253  // of the near-plane is (0,0,-1,1) (maybe I have to
254  // switch around the < 0 checks in the function?)

354 template <typename T, glm::precision P = glm::defaultp>
355 std::vector<detail::Vertex<T, P>>
356 clip_polygon_to_plane_in_4d(const std::vector<detail::Vertex<T, P>>& vertices,
357  const glm::tvec4<T, P>& plane_normal)
358 {
359  std::vector<detail::Vertex<T, P>> clipped_vertices;
360 
361  // We can have 2 cases:
362  // * 1 vertex visible: we make 1 new triangle out of the visible vertex plus the 2 intersection points
363  // with the near-plane
364  // * 2 vertices visible: we have a quad, so we have to make 2 new triangles out of it.
365 
366  // See here for more info?
367  // http://math.stackexchange.com/questions/400268/equation-for-a-line-through-a-plane-in-homogeneous-coordinates
368 
369  for (unsigned int i = 0; i < vertices.size(); i++)
370  {
371  int a = i; // the current vertex
372  int b = (i + 1) % vertices.size(); // the following vertex (wraps around 0)
373 
374  using glm::dot;
375  const T fa = dot(vertices[a].position, plane_normal); // Note: Shouldn't they be unit length?
376  const T fb = dot(vertices[b].position,
377  plane_normal); // < 0 means on visible side, > 0 means on invisible side?
378 
379  if ((fa < 0 && fb > 0) || (fa > 0 && fb < 0)) // one vertex is on the visible side of the plane, one
380  // on the invisible? so we need to split?
381  {
382  const glm::tvec4<T, P> direction = vertices[b].position - vertices[a].position;
383  const T t = -(dot(plane_normal, vertices[a].position)) /
384  (dot(plane_normal, direction)); // the parametric value on the line, where the line to
385  // draw intersects the plane?
386 
387  // generate a new vertex at the line-plane intersection point
388  const glm::tvec4<T, P> position = vertices[a].position + t * direction;
389  const glm::tvec3<T, P> color = vertices[a].color + t * (vertices[b].color - vertices[a].color);
390  const glm::tvec2<T, P> texcoords =
391  vertices[a].texcoords +
392  t * (vertices[b].texcoords -
393  vertices[a].texcoords); // We could omit that if we don't render with texture.
394 
395  if (fa < 0) // we keep the original vertex plus the new one
396  {
397  clipped_vertices.push_back(vertices[a]);
398  clipped_vertices.push_back(detail::Vertex<T, P>{position, color, texcoords});
399  } else if (fb < 0) // we use only the new vertex
400  {
401  clipped_vertices.push_back(detail::Vertex<T, P>{position, color, texcoords});
402  }
403  } else if (fa < 0 && fb < 0) // both are visible (on the "good" side of the plane), no splitting
404  // required, use the current vertex
405  {
406  clipped_vertices.push_back(vertices[a]);
407  }
408  // else, both vertices are not visible, nothing to add and draw
409  }
410 
411  return clipped_vertices;
412 };
