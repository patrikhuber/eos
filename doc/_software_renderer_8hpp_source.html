97 template <typename T>
98 core::Image4u render(const core::Mesh& mesh, const Eigen::Matrix4<T>& model_view_matrix,
99 const Eigen::Matrix4<T>& projection_matrix,
100 const cpp17::optional<Texture>& texture = cpp17::nullopt)
101 {
102 // The number of vertices has to be equal for both shape and colour, or, alternatively, it has to be a
103 // shape-only model:
104 assert(mesh.vertices.size() == mesh.colors.size() || mesh.colors.empty());
105 // Make sure one of these three things is true:
106 // a) There are texture coordinates given for each vertex (the texture map doesn't contain any seams)
107 // b) A different number of tex coords and vertices (i.e. the texture map contains seams): A separate
108 // list of texture triangle indices has to be given (i.e. mesh.tti is not empty)
109 // c) There are no texture coordinates given (i.e. rendering without texture).
110 assert(mesh.vertices.size() == mesh.texcoords.size() ||
111 ((mesh.vertices.size() != mesh.texcoords.size()) && !mesh.tti.empty()) ||
112 mesh.texcoords.empty());
113 // Sanity check on the texture triangle indices: They should be either empty or equal to tvi.size():
114 assert(mesh.tti.empty() || mesh.tti.size() == mesh.tvi.size());
115 // Make sure either no texture is given, or a texture and texture coords:
116 // (Potential todo: Is there a texturing flag we should check for too?)
117 assert(!texture.has_value() || (texture.has_value() && mesh.texcoords.size() > 0));
118
119 using detail::divide_by_w;
120 using std::vector;
121
122 vector<Eigen::Vector4<T>> clipspace_vertices;
123 for (const auto& vertex_position : mesh.vertices)
124 {
125 // Note: We make an unnecessary copy of the vertex_position here. We could perhaps improve this by
126 // using vertex_shader.operator()<vertex_type, T>(...). We're also making vertex_position the same
127 // type as the matrix here, which we wouldn't necessarily need to, as VertexShader supports a
128 // VertexType and MatrixType.
129 clipspace_vertices.push_back(
130 vertex_shader(Eigen::Vector4<T>(vertex_position.homogeneous()), model_view_matrix, projection_matrix));
131 // Note: if mesh.colors.empty() (in case of shape-only model!), then the vertex colour is no
132 // longer set to gray. But we don't want that here, maybe we only want texturing, then we don't
133 // need vertex-colours at all. We can do it in a custom VertexShader if needed.
134 }
135
136 // All vertices are in clip-space now. Prepare the rasterisation stage:
137 vector<Triangle<T>> triangles_to_raster;
138 const auto& tti = mesh.tti.empty() ? mesh.tvi : mesh.tti; // If tti is not empty, we'll use it, otherwise,
139 // use tvi for texturing.
140
141 // In the loop further below, we're building a list of triangles to render, and will be indexing into
142 // Mesh::colors. But that array is not available for meshes without per-vertex colouring. So we'll use
143 // the below lambda which either returns the colours for the vertex, or an empty vector.
144 const auto get_color_or_zero = [](const std::vector<Eigen::Vector3f>& mesh_colors, int vertex_index) {
145 if (mesh_colors.size() > 0)
146 {
147 return mesh_colors[vertex_index];
148 } else
149 {
150 return Eigen::Vector3f(0.0f, 0.0f, 0.0f);
151 }
152 };
153
154 // This builds the (one and final) triangles to render. Meaning: The triangles formed of mesh.tvi (the
155 // ones that survived the clip/culling), plus possibly more that intersect one of the frustum planes
156 // (i.e. this can generate new triangles with new pos/vc/texcoords).
157 for (std::size_t tri_index = 0; tri_index < mesh.tvi.size(); ++tri_index)
158 {
159 const auto& tri_indices = mesh.tvi[tri_index];
160 const auto& tri_tc_indices = tti[tri_index];
161 unsigned char visibility_bits[3];
162 for (unsigned char k = 0; k < 3; k++)
163 {
164 visibility_bits[k] = 0;
165 const auto x_cc = clipspace_vertices[tri_indices[k]].x();
166 const auto y_cc = clipspace_vertices[tri_indices[k]].y();
167 const auto z_cc = clipspace_vertices[tri_indices[k]].z();
168 const auto w_cc = clipspace_vertices[tri_indices[k]].w();
169 if (x_cc < -w_cc) // true if outside of view frustum. False if on or inside the plane.
170 visibility_bits[k] |= 1; // set bit if outside of frustum
171 if (x_cc > w_cc)
172 visibility_bits[k] |= 2;
173 if (y_cc < -w_cc)
174 visibility_bits[k] |= 4;
175 if (y_cc > w_cc)
176 visibility_bits[k] |= 8;
177 if (enable_near_clipping && z_cc < -w_cc) // near plane frustum clipping
178 visibility_bits[k] |= 16;
179 if (rasterizer.enable_far_clipping && z_cc > w_cc) // far plane frustum clipping
180 visibility_bits[k] |= 32;
181 } // if all bits are 0, then it's inside the frustum
182 // all vertices are not visible - reject the triangle.
183 if ((visibility_bits[0] & visibility_bits[1] & visibility_bits[2]) > 0)
184 {
185 continue;
186 }
187 // all vertices are visible - pass the whole triangle to the rasteriser. = All bits of all 3
188 // triangles are 0.
189 if ((visibility_bits[0] | visibility_bits[1] | visibility_bits[2]) == 0)
190 {
191 // relevant part of process_prospective_tri:
192 std::array<Eigen::Vector4<T>, 3> prospective_tri{
193 divide_by_w(clipspace_vertices[tri_indices[0]]),
194 divide_by_w(clipspace_vertices[tri_indices[1]]),
195 divide_by_w(clipspace_vertices[tri_indices[2]])};
196 // We have a prospective tri in NDC coords now, with its vertices having coords [x_ndc, y_ndc,
197 // z_ndc, 1/w_clip].
198
199 // Replaces x and y of the NDC coords with the screen coords. Keep z and w the same.
200 const Eigen::Vector2<T> v0_screen =
201 clip_to_screen_space(prospective_tri[0].x(), prospective_tri[0].y(),
202 rasterizer.viewport_width, rasterizer.viewport_height);
203 prospective_tri[0].x() = v0_screen.x();
204 prospective_tri[0].y() = v0_screen.y();
205 const Eigen::Vector2<T> v1_screen =
206 clip_to_screen_space(prospective_tri[1].x(), prospective_tri[1].y(),
207 rasterizer.viewport_width, rasterizer.viewport_height);
208 prospective_tri[1].x() = v1_screen.x();
209 prospective_tri[1].y() = v1_screen.y();
210 const Eigen::Vector2<T> v2_screen =
211 clip_to_screen_space(prospective_tri[2].x(), prospective_tri[2].y(),
212 rasterizer.viewport_width, rasterizer.viewport_height);
213 prospective_tri[2].x() = v2_screen.x();
214 prospective_tri[2].y() = v2_screen.y();
215
216 // Culling (front/back/none - or what are OpenGL's modes?). Do we do any culling
217 // elsewhere? No?
218 if (enable_backface_culling)
219 {
220 // Note/Todo: Isn't this just v0_screen, etc.? Let's assume for now it is. So before, we had here:
221 // prospective_tri[0], [1], [2], but we only need their x and y coords.
222 if (!detail::are_vertices_ccw_in_screen_space(
223 Eigen::Vector2<T>(prospective_tri[0].x(), prospective_tri[0].y()),
224 Eigen::Vector2<T>(prospective_tri[1].x(), prospective_tri[1].y()),
225 Eigen::Vector2<T>(prospective_tri[2].x(), prospective_tri[2].y())))
226 continue;
227 }
228
229 // Get the bounding box of the triangle:
230 const auto boundingBox = detail::calculate_clipped_bounding_box(
231 Eigen::Vector2<T>(prospective_tri[0].x(), prospective_tri[0].y()),
232 Eigen::Vector2<T>(prospective_tri[1].x(), prospective_tri[1].y()),
233 Eigen::Vector2<T>(prospective_tri[2].x(), prospective_tri[2].y()),
234 rasterizer.viewport_width, rasterizer.viewport_height);
235 const auto min_x = boundingBox.x;
236 const auto max_x = boundingBox.x + boundingBox.width;
237 const auto min_y = boundingBox.y;
238 const auto max_y = boundingBox.y + boundingBox.height;
239 if (max_x <= min_x || max_y <= min_y)
240 { // Note: Can the width/height of the bbox be negative? Maybe we only need to check for
241 // equality here?
242 continue;
243 }
244
245 // If we're here, the triangle is CCW in screen space and the bbox is inside the viewport!
246 triangles_to_raster.push_back(Triangle<T>{
247 detail::Vertex<T>{prospective_tri[0], get_color_or_zero(mesh.colors, tri_indices[0]),
248 Eigen::Vector2<T>(mesh.texcoords[tri_tc_indices[0]](0),
249 mesh.texcoords[tri_tc_indices[0]](1))},
250 detail::Vertex<T>{prospective_tri[1], get_color_or_zero(mesh.colors, tri_indices[1]),
251 Eigen::Vector2<T>(mesh.texcoords[tri_tc_indices[1]](0),
252 mesh.texcoords[tri_tc_indices[1]](1))},
253 detail::Vertex<T>{prospective_tri[2], get_color_or_zero(mesh.colors, tri_indices[2]),
254 Eigen::Vector2<T>(mesh.texcoords[tri_tc_indices[2]](0),
255 mesh.texcoords[tri_tc_indices[2]](1))}});
256 continue; // Triangle was either added or not added. Continue with next triangle.
257 }
258 // At this point, the triangle is known to be intersecting one of the view frustum's planes
259 // Note: It seems that this is only w.r.t. the near-plane. If a triangle is partially outside the
260 // tlbr viewport, it'll get rejected.
261 // Well, 'z' of these triangles seems to be -1, so is that really the near plane?
262 std::vector<detail::Vertex<T>> vertices;
263 vertices.reserve(3);
264 vertices.push_back(detail::Vertex<T>{clipspace_vertices[tri_indices[0]],
265 get_color_or_zero(mesh.colors, tri_indices[0]),
266 Eigen::Vector2<T>(mesh.texcoords[tri_tc_indices[0]](0),
267 mesh.texcoords[tri_tc_indices[0]](1))});
268 vertices.push_back(detail::Vertex<T>{clipspace_vertices[tri_indices[1]],
269 get_color_or_zero(mesh.colors, tri_indices[1]),
270 Eigen::Vector2<T>(mesh.texcoords[tri_tc_indices[1]](0),
271 mesh.texcoords[tri_tc_indices[1]](1))});
272 vertices.push_back(detail::Vertex<T>{clipspace_vertices[tri_indices[2]],
273 get_color_or_zero(mesh.colors, tri_indices[2]),
274 Eigen::Vector2<T>(mesh.texcoords[tri_tc_indices[2]](0),
275 mesh.texcoords[tri_tc_indices[2]](1))});
276 // split the triangle if it intersects the near plane:
277 if (enable_near_clipping)
278 {
279 vertices = clip_polygon_to_plane_in_4d(
280 vertices, Eigen::Vector4<T>(T(0.0), T(0.0), T(-1.0),
281 T(-1.0))); // "Normal" (or "4D hyperplane") of the near-plane.
282 // I tested it and it works like this but I'm a
283 // little bit unsure because Songho says the normal
284 // of the near-plane is (0,0,-1,1) (maybe I have to
285 // switch around the < 0 checks in the function?)
286 }
287
288 // Triangulation of the polygon formed of the 'vertices' array:
289 if (vertices.size() >= 3)
290 {
291 for (unsigned char k = 0; k < vertices.size() - 2; k++)
292 {
293 // Build a triangle from vertices[0], vertices[1 + k], vertices[2 + k]:
294 // Add to triangles_to_raster if it passed culling etc.
295 // TODO: This does the same as above - the code is copied 1:1. Avoid!
296 // COPY START (but init of prospective_tri is changed, as well as init of 't'.)
297 std::array<Eigen::Vector4<T>, 3> prospective_tri{divide_by_w(vertices[0].position),
298 divide_by_w(vertices[1 + k].position),
299 divide_by_w(vertices[2 + k].position)};
300
301 const Eigen::Vector2<T> v0_screen =
302 clip_to_screen_space(prospective_tri[0].x(), prospective_tri[0].y(),
303 rasterizer.viewport_width, rasterizer.viewport_height);
304 prospective_tri[0].x() = v0_screen.x();
305 prospective_tri[0].y() = v0_screen.y();
306 const Eigen::Vector2<T> v1_screen =
307 clip_to_screen_space(prospective_tri[1].x(), prospective_tri[1].y(),
308 rasterizer.viewport_width, rasterizer.viewport_height);
309 prospective_tri[1].x() = v1_screen.x();
310 prospective_tri[1].y() = v1_screen.y();
311 const Eigen::Vector2<T> v2_screen =
312 clip_to_screen_space(prospective_tri[2].x(), prospective_tri[2].y(),
313 rasterizer.viewport_width, rasterizer.viewport_height);
314 prospective_tri[2].x() = v2_screen.x();
315 prospective_tri[2].y() = v2_screen.y();
316
317 if (enable_backface_culling)
318 {
319 if (!detail::are_vertices_ccw_in_screen_space(
320 Eigen::Vector2<T>(prospective_tri[0].x(), prospective_tri[0].y()),
321 Eigen::Vector2<T>(prospective_tri[1].x(), prospective_tri[1].y()),
322 Eigen::Vector2<T>(prospective_tri[2].x(), prospective_tri[2].y())))
323 continue;
324 }
325
326 const auto boundingBox = detail::calculate_clipped_bounding_box(
327 Eigen::Vector2<T>(prospective_tri[0].x(), prospective_tri[0].y()),
328 Eigen::Vector2<T>(prospective_tri[1].x(), prospective_tri[1].y()),
329 Eigen::Vector2<T>(prospective_tri[2].x(), prospective_tri[2].y()),
330 rasterizer.viewport_width, rasterizer.viewport_height);
331 const auto min_x = boundingBox.x;
332 const auto max_x = boundingBox.x + boundingBox.width;
333 const auto min_y = boundingBox.y;
334 const auto max_y = boundingBox.y + boundingBox.height;
335 if (max_x <= min_x || max_y <= min_y)
336 {
337 continue;
338 }
339
340 // If we're here, the triangle is CCW in screen space and the bbox is inside the viewport!
341 triangles_to_raster.push_back(Triangle<T>{
342 detail::Vertex<T>{prospective_tri[0], vertices[0].color, vertices[0].texcoords},
343 detail::Vertex<T>{prospective_tri[1], vertices[1 + k].color,
344 vertices[1 + k].texcoords},
345 detail::Vertex<T>{prospective_tri[2], vertices[2 + k].color,
346 vertices[2 + k].texcoords}});
347 // continue; // triangle was either added or not added. Continue with next triangle.
348 // COPY END
349 }
350 }
351
352 } // end of loop over all triangles
353
354 // Each triangle contains [x_screen, y_screen, z_ndc, 1/w_clip].
355 // We may have more triangles than in the original mesh.
356
357 // Raster each triangle and apply the fragment shader on each pixel:
358 for (const auto& tri : triangles_to_raster)
359 {
360 rasterizer.raster_triangle(tri[0], tri[1], tri[2], texture);
361 }
362 return rasterizer.colorbuffer;
363 };
